155 void initializeFromStudy(Data::Study& study)
156 {
157 pNbYearsParallel = study.maxNbYearsInParallel;
158
159 yearsWeight = study.parameters.getYearsWeight();
160 yearsWeightSum = study.parameters.getYearsWeightSum();
161
162 // Average on all years
163 for (uint i = 0; i != VCardType::columnCount; ++i)
164 {
165 AncestorType::pResults[i].initializeFromStudy(study);
166 AncestorType::pResults[i].reset();
167 }
168
169 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
170 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; ++numSpace)
171 for (unsigned int i = 0; i != VCardType::columnCount; ++i)
172 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(study);
173
174 pValuesForYearLocalReport = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
175 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; ++numSpace)

273 void hourForEachLink(State& state, unsigned int numSpace)
274 {
275 // Ratio take into account MC year weight
276 float ratio = yearsWeight[state.year] / yearsWeightSum;
277
278 assert(state.link != NULL);
279 const auto& linkDirectCapa = state.link->directCapacities;
280 const auto& linkIndirectCapa = state.link->indirectCapacities;
281 const int tsIndex = NumeroChroniquesTireesParInterconnexion[numSpace][state.link->index]
282 .TransmissionCapacities;
283 // CONG. PROB +
284 if (state.ntc.ValeurDuFlux[state.link->index]
285 > +linkDirectCapa.entry[tsIndex][state.hourInTheYear] - 10e-6)
286 pValuesForTheCurrentYear[numSpace][0].hour[state.hourInTheYear] += 100.0 * ratio;
287 // CONG. PROB -
288 if (state.ntc.ValeurDuFlux[state.link->index]
289 < -linkIndirectCapa.entry[tsIndex][state.hourInTheYear] + 10e-6)
290 pValuesForTheCurrentYear[numSpace][1].hour[state.hourInTheYear] += 100.0 * ratio;
291
292 // Next item in the list
293 NextType::hourForEachLink(state, numSpace);
294 }

296 void beforeYearByYearExport(uint /*year*/, uint numSpace)
297 {
298 for (uint i = 0; i != VCardType::columnCount; ++i)
299 {
300 for (uint h = 0; h != maxHoursInAYear; ++h)
301 pValuesForYearLocalReport[numSpace][i].hour[h]
302 = (pValuesForTheCurrentYear[numSpace][i].hour[h] > 0.) ? 100. : 0.;
303
304 for (uint d = 0; d != maxDaysInAYear; ++d)
305 pValuesForYearLocalReport[numSpace][i].day[d]
306 = (pValuesForTheCurrentYear[numSpace][i].day[d] > 0.) ? 100. : 0.;
307
308 for (uint w = 0; w != maxWeeksInAYear; ++w)
309 pValuesForYearLocalReport[numSpace][i].week[w]
310 = (pValuesForTheCurrentYear[numSpace][i].week[w] > 0.) ? 100. : 0.;
311
312 for (uint m = 0; m != maxMonths; ++m)
313 pValuesForYearLocalReport[numSpace][i].month[m]
314 = (pValuesForTheCurrentYear[numSpace][i].month[m] > 0.) ? 100. : 0.;
315
316 pValuesForYearLocalReport[numSpace][i].year
317 = (pValuesForTheCurrentYear[numSpace][i].year > 0.) ? 100. : 0.;
318 }
