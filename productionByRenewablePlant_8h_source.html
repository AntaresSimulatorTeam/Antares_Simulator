155 void initializeFromArea(Data::Study* study, Data::Area* area)
156 {
157 // Get the number of years in parallel
158 pNbYearsParallel = study->maxNbYearsInParallel;
159 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
160
161 // Get the area
162 pSize = area->renewable.clusterCount();
163 if (pSize)
164 {
165 AncestorType::pResults.resize(pSize);
166
167 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
168 pValuesForTheCurrentYear[numSpace]
169 = new VCardType::IntermediateValuesDeepType[pSize];
170
171 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
172 for (unsigned int i = 0; i != pSize; ++i)
173 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
174
175 for (unsigned int i = 0; i != pSize; ++i)
176 {
177 AncestorType::pResults[i].initializeFromStudy(*study);
178 AncestorType::pResults[i].reset();
179 }
180 }
181 else
182 {
183 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
184 {
185 pValuesForTheCurrentYear[numSpace] = nullptr;
186 }
187
188 AncestorType::pResults.clear();
189 }
190 // Next
191 NextType::initializeFromArea(study, area);
192 }

278 void hourForEachArea(State& state, unsigned int numSpace)
279 {
280 for (uint clusterIndex = 0; clusterIndex != state.area->renewable.clusterCount(); ++clusterIndex)
281 {
282 const auto* renewableCluster = state.area->renewable.clusters[clusterIndex];
283 uint serieIndex = state.timeseriesIndex->RenouvelableParPalier[clusterIndex];
284 double renewableClusterProduction = renewableCluster->valueAtTimeStep(serieIndex, state.hourInTheYear);
285
286 pValuesForTheCurrentYear[numSpace][renewableCluster->areaWideIndex].hour[state.hourInTheYear]
287 += renewableClusterProduction;
288 }
289
290 // Next variable
291 NextType::hourForEachArea(state, numSpace);
292 }
