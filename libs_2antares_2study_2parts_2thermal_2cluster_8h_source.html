304 // v3.4:

306 //
307 // v3.5:
308 // solver input : MB
309 // output :
310 // O(h) = MA * P(h)
311 // if (P(h) > 0)
312 // O(h) += FI

316 // \see 101206-antares-couts.doc
317 //
318 // v5.0:
319 // Abs( SU ) = 0 or in [0.005;5000000]
