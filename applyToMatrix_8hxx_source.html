48template<class StringT, class D>
49bool ApplyToMatrix(uint& errors,
50 StringT& logprefix,
51 D& data,
52 const TSNumberData::MatrixType::ColumnType& years,
53 uint tsGenMax)
54{
55 bool ret = true;
56
57 // In this case, m.height represents the total number of years
58 const uint nbYears = data.timeseriesNumbers.height;
59 // The matrix m has only one column
60 assert(data.timeseriesNumbers.width == 1);
61 typename Matrix<uint32_t>::ColumnType& target = data.timeseriesNumbers[0];
62
63 for (uint y = 0; y != nbYears; ++y)
64 {
65 if (years[y] != 0)
66 {
67 // The new TS number
68 uint tsNum = years[y] - 1;
69
70 // When the TS-Generators are not used
71 if (!CheckValidity(tsNum, data, tsGenMax))
72 {
73 if (errors <= maxErrors)
74 {
75 if (++errors == maxErrors)
76 logs.warning() << "scenario-builder: ... (skipped)";
77 else
78 logs.warning() << "scenario-builder: " << logprefix
79 << "value out of bounds for the year " << (y + 1);
80 }
81 ret = false;
82 continue;
83 }
84 // Ok, assign. The value provided by the interface is user-friendly
85 // and starts from 1.
86 target[y] = tsNum;
87 }
88 }
89
90 return ret;
91}
