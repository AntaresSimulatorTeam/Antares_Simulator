144 void initializeFromStudy(Data::Study& study)
145 {
146 pNbYearsParallel = study.maxNbYearsInParallel;
147
148 // Average on all years
149 InitializeResultsFromStudy(AncestorType::pResults, study);
150
151 // Intermediate values
152 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
153 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
154 pValuesForTheCurrentYear[numSpace].initializeFromStudy(study);
155
156 // Next
157 NextType::initializeFromStudy(study);
158 }
159
160 void initializeFromArea(Data::Study* study, Data::Area* area)
161 {
162 // Copy raw values
163 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
164 {
165 // Memcpy for the first value
166 (void)::memcpy(pValuesForTheCurrentYear[numSpace].hour,
167 area->miscGen.entry[0],
168 sizeof(double) * area->miscGen.height);
169
170 {
171 // We must not include the CHP and the Solar. So we have to start from `2`
172 for (unsigned int x = 1; x != (unsigned int)Data::fhhPSP; ++x)
173 {
174 const Matrix<>::ColumnType& col = area->miscGen.entry[x];
175 for (unsigned int y = 0; y != area->miscGen.height; ++y)
176 pValuesForTheCurrentYear[numSpace].hour[y] += col[y];
177 }
178 }
179 }
180
181 // Next
182 NextType::initializeFromArea(study, area);
183 }

187 // Next
188 NextType::initializeFromAreaLink(study, link);
