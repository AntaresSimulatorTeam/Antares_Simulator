65template<class VCardT>
66inline void IntermediateValues::buildAnnualSurveyReport(SurveyResults& report,
67 int fileLevel,
68 int precision) const
69{
70 if (!(fileLevel & Category::id))
71 {
72 switch (precision)
73 {
74 case Category::hourly:
75 internalExportAnnualValues<maxHoursInAYear, VCardT>(report, hour, false);
76 break;
77 case Category::daily:
78 internalExportAnnualValues<maxDaysInAYear, VCardT>(report, day, false);
79 break;
80 case Category::weekly:
81 internalExportAnnualValues<maxWeeksInAYear, VCardT>(report, week, false);
82 break;
83 case Category::monthly:
84 internalExportAnnualValues<maxMonths, VCardT>(report, month, false);
85 break;
86 case Category::annual:
87 internalExportAnnualValues<1, VCardT>(report, &year, true);
88 break;
89 }
90 }
91}

93template<unsigned int Size, class VCardT, class A>
94void IntermediateValues::internalExportAnnualValues(SurveyResults& report,
95 const A& array,
96 bool annual) const
97{
98 using namespace Yuni;
99 assert(report.data.columnIndex < report.maxVariables && "Column index out of bounds");
100
101 // Caption
102 report.captions[0][report.data.columnIndex] = report.variableCaption;
103 report.captions[1][report.data.columnIndex] = report.variableUnit;
104 report.captions[2][report.data.columnIndex] = nullptr;
105 // Precision
106 report.precision[report.data.columnIndex] = PrecisionToPrintfFormat<VCardT::decimal>::Value();
107 // Non applicability
108 report.nonApplicableStatus[report.data.columnIndex] = *report.isCurrentVarNA;
109
110 // Values
111 if (not annual)
112 {
113 (void)::memcpy(report.values[report.data.columnIndex], array, sizeof(double) * Size);
114 }
115 else
116 {
117 double& target = *(report.values[report.data.columnIndex]);
118 target = year;
119 }
120
121 // Next column index
122 ++report.data.columnIndex;
123}
