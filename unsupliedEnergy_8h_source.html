104template<class NextT = Container::EndOfList>
105class UnsupliedEnergy
106 : public Variable::IVariable<UnsupliedEnergy<NextT>, NextT, VCardUnsupliedEnergy>
107{
108public:
110 typedef NextT NextType;
112 typedef VCardUnsupliedEnergy VCardType;
114 typedef Variable::IVariable<UnsupliedEnergy<NextT>, NextT, VCardType> AncestorType;
115
117 typedef typename VCardType::ResultsType ResultsType;
118
119 typedef VariableAccessor<ResultsType, VCardType::columnCount> VariableAccessorType;

121 enum
122 {
124 count = 1 + NextT::count,

127 template<int CDataLevel, int CFile>
128 struct Statistics
129 {
130 enum
131 {
132 count
133 = ((VCardType::categoryDataLevel & CDataLevel && VCardType::categoryFileLevel & CFile)
134 ? (NextType::template Statistics<CDataLevel, CFile>::count
135 + VCardType::columnCount * ResultsType::count)
136 : NextType::template Statistics<CDataLevel, CFile>::count),
137 };
138 };

140public:
141 ~UnsupliedEnergy()
142 {
143 delete[] pValuesForTheCurrentYear;
144 }
145
146 void initializeFromStudy(Data::Study& study)
147 {
148 pNbYearsParallel = study.maxNbYearsInParallel;
149
150 // Intermediate values
151 InitializeResultsFromStudy(AncestorType::pResults, study);
152
153 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
154 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
155 pValuesForTheCurrentYear[numSpace].initializeFromStudy(study);
156
157 // Next
158 NextType::initializeFromStudy(study);
159 }
160
161 template<class R>
162 static void InitializeResultsFromStudy(R& results, Data::Study& study)
163 {
164 VariableAccessorType::InitializeAndReset(results, study);
165 }
166
167 void initializeFromArea(Data::Study* study, Data::Area* area)
168 {
169 // Next
170 NextType::initializeFromArea(study, area);
171 }
172
173 void initializeFromLink(Data::Study* study, Data::AreaLink* link)
174 {
175 // Next
176 NextType::initializeFromAreaLink(study, link);
177 }
178
179 void simulationBegin()
180 {
181 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
182 pValuesForTheCurrentYear[numSpace].reset();
183 // Next
184 NextType::simulationBegin();
185 }
186
187 void simulationEnd()
188 {
189 NextType::simulationEnd();
190 }

192 void yearBegin(unsigned int year, unsigned int numSpace)
193 {
194 // Reset the values for the current year
195 pValuesForTheCurrentYear[numSpace].reset();
196
197 // Next variable
198 NextType::yearBegin(year, numSpace);
199 }
200
201 void yearEndBuild(State& state, unsigned int year)
202 {
203 // Next variable
204 NextType::yearEndBuild(state, year);
205 }
206
207 void yearEnd(unsigned int year, unsigned int numSpace)
208 {
209 // Compute all statistics for the current year (daily,weekly,monthly)
210 pValuesForTheCurrentYear[numSpace].computeStatisticsForTheCurrentYear();
211
212 // Next variable
213 NextType::yearEnd(year, numSpace);
214 }
215
216 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
217 unsigned int nbYearsForCurrentSummary)
218 {
219 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
220 {
221 // Merge all those values with the global results
222 AncestorType::pResults.merge(numSpaceToYear[numSpace] /*year*/,
223 pValuesForTheCurrentYear[numSpace]);
224 }
225
226 // Next variable
227 NextType::computeSummary(numSpaceToYear, nbYearsForCurrentSummary);
228 }
229
230 void hourBegin(unsigned int hourInTheYear)
231 {
232 // Next variable
233 NextType::hourBegin(hourInTheYear);
234 }
235
236 void hourForEachArea(State& state, unsigned int numSpace)
237 {
238 // Total UnsupliedEnergy emissions
239 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] =
240 // Current Hydro Storage generation
241 state.hourlyResults->ValeursHorairesDeDefaillancePositive[state.hourInTheWeek];
242
243 // Next variable
244 NextType::hourForEachArea(state, numSpace);
245 }
246
247 Antares::Memory::Stored<double>::ConstReturnType retrieveRawHourlyValuesForCurrentYear(
248 unsigned int,

251 return pValuesForTheCurrentYear[numSpace].hour;
252 }

254 void localBuildAnnualSurveyReport(SurveyResults& results,
255 int fileLevel,
256 int precision,
257 unsigned int numSpace) const
258 {
259 // Initializing external pointer on current variable non applicable status
260 results.isCurrentVarNA = AncestorType::isNonApplicable;
261
262 if (AncestorType::isPrinted[0])
263 {
264 // Write the data for the current year
265 results.variableCaption = VCardType::Caption();
266 results.variableUnit = VCardType::Unit();
267 pValuesForTheCurrentYear[numSpace].template buildAnnualSurveyReport<VCardType>(
268 results, fileLevel, precision);
269 }
270 }
