341class computeTimeStepLevel
342{
343private:
344 int step;
345 double level;
346
347 double capacity;
348 std::vector<double>& inflows;
349 std::vector<double>& ovf;
350 std::vector<double>& turb;
351 double pumpRatio;
352 std::vector<double>& pump;
353 double excessDown;
354
355public:
356 computeTimeStepLevel(
357 const double& startLvl,
358 std::vector<double>& infl,
359 std::vector<double>& overfl,
360 std::vector<double>& H,
361 double pumpEff,
362 std::vector<double>& Pump,
363 double rc) :
364 step(0),
365 level(startLvl),
366 capacity(rc),
367 inflows(infl),
368 ovf(overfl),
369 turb(H),
370 pumpRatio(pumpEff),
371 pump(Pump),
372 excessDown(0.)
373 {
374 }
375
376 void run()
377 {
378 excessDown = 0.;
379
380 level = level + inflows[step] - turb[step] + pumpRatio * pump[step];
381
382 if (level > capacity)
383 {
384 ovf[step] = level - capacity;
385 level = capacity;
386 }
387
388 if (level < 0)
389 {
390 excessDown = -level;
391 level = 0.;
392 inflows[step] += excessDown;
393 }
394 }
395
396 void prepareNextStep()
397 {
398 step++;
399
400 inflows[step] -= excessDown;
401 }
402
403 double getLevel()
404 {
405 return level;
406 }
407};
