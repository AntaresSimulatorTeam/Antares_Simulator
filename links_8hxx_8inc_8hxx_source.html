168inline void Links::buildSurveyReport(SurveyResults& results,
169 int dataLevel,
170 int fileLevel,
171 int precision) const
172{
173 int count_int = count;
174 bool link_dataLevel = (dataLevel & Category::link);
175 if (count_int && link_dataLevel)
176 {
177 assert(results.data.link != NULL
178 && "The link must not be null to generate a survey report");
179
180 // Build the survey results for the given area
181 auto& link = *results.data.link;
182
183 // Filtering
184 switch (precision)
185 {
186 case Category::hourly:
187 if (!(link.filterSynthesis & Data::filterHourly))
188 return;
189 break;
190 case Category::daily:
191 if (!(link.filterSynthesis & Data::filterDaily))
192 return;
193 break;
194 case Category::weekly:
195 if (!(link.filterSynthesis & Data::filterWeekly))
196 return;
197 break;
198 case Category::monthly:
199 if (!(link.filterSynthesis & Data::filterMonthly))
200 return;
201 break;
202 case Category::annual:
203 if (!(link.filterSynthesis & Data::filterAnnual))
204 return;
205 break;
206 case Category::all:
207 break;
208 }
209 auto& linkOutputData = pLinks[link.indexForArea];
210 linkOutputData.buildSurveyReport(results, dataLevel, fileLevel, precision);
211 }
212}

214inline void Links::buildAnnualSurveyReport(SurveyResults& results,
215 int dataLevel,
216 int fileLevel,
217 int precision,
218 uint numSpace) const
219{
220 int count_int = count;
221 bool link_dataLevel = (dataLevel & Category::link);
222 if (count_int && link_dataLevel)
223 {
224 assert(results.data.link != NULL
225 && "The link must not be null to generate a survey report");
226
227 auto& link = *results.data.link;
228
229 // Filtering
230 switch (precision)
231 {
232 case Category::hourly:
233 if (not(link.filterYearByYear & Data::filterHourly))
234 return;
235 break;
236 case Category::daily:
237 if (not(link.filterYearByYear & Data::filterDaily))
238 return;
239 break;
240 case Category::weekly:
241 if (not(link.filterYearByYear & Data::filterWeekly))
242 return;
243 break;
244 case Category::monthly:
245 if (not(link.filterYearByYear & Data::filterMonthly))
246 return;
247 break;
248 case Category::annual:
249 if (not(link.filterYearByYear & Data::filterAnnual))
250 return;
251 break;
252 case Category::all:
253 break;
254 }
255
256 // Build the survey results for the given area
257 auto& linkOutputData = pLinks[results.data.link->indexForArea];
258 linkOutputData.buildAnnualSurveyReport(results, dataLevel, fileLevel, precision, numSpace);
259 }

262template<class I>
263inline void Links::provideInformations(I& infos)
264{
265 // Begining of the node
266 if (VCardType::nodeDepthForGUI)
267 {
268 infos.template beginNode<VCardType>();
269 // Next variable in the list
270 NextType::template provideInformations<I>(infos);
271 // End of the node
272 infos.endNode();
273 }
274 else
275 {
276 // Giving our VCard
277 infos.template addVCard<VCardType>();
278 // Next variable in the list
279 NextType::template provideInformations<I>(infos);
280 }
