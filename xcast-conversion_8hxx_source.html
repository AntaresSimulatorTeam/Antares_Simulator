105template<enum Data::TimeSeries T>
106bool XCastConversion<T>::onMatrixResize(uint, uint, uint& newX, uint&)
107{
108 if (newX > Data::XCast::conversionMaxPoints)
109 return false;
110 auto& matrix = (*pMatrix);
111 matrix[matrix.width - 1][0] = 0;
112 matrix[matrix.width - 1][1] = 0;
113 matrix.resizeWithoutDataLost(newX + 2, 2);
114 matrix[matrix.width - 1][0] = +std::numeric_limits<float>::max();
115 matrix.markAsModified();
116 MarkTheStudyAsModified();
117 return false;
118}

120template<enum Data::TimeSeries T>
121void XCastConversion<T>::onMatrixLoad()
122{
123 auto& matrix = *pMatrix;
124 if (matrix.height != 2)
125 {
126 // It may happen when the file is not loaded
127 matrix.reset(3, 2);
128 matrix.markAsModified();
129 MarkTheStudyAsModified();
130 }
131 if (matrix.width >= 3 && matrix.width <= Data::XCast::conversionMaxPoints)
132 {
133 matrix[0][0] = -std::numeric_limits<float>::max();
134 matrix[0][1] = matrix[1][1];
135 matrix[matrix.width - 1][0] = +std::numeric_limits<float>::max();
136 matrix[matrix.width - 1][1] = matrix[matrix.width - 2][1];
137 }
138 else
139 {
140 matrix.reset(3, 2);
141 matrix[0][0] = -std::numeric_limits<float>::max();
142 matrix[0][1] = 0.f;
143 matrix[1][0] = 0.f;
144 matrix[1][1] = 0.f;
145 matrix[2][0] = +std::numeric_limits<float>::max();
146 matrix[2][1] = 0.f;
147 }
148}
