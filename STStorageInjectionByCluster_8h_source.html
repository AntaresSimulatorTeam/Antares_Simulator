144 void initializeFromArea(Data::Study* study, Data::Area* area)
145 {
146 // Get the number of years in parallel
147 pNbYearsParallel = study->maxNbYearsInParallel;
148 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
149
150 // Get the area
151 nbClusters_ = area->shortTermStorage.count();
152 if (nbClusters_)
153 {
154 AncestorType::pResults.resize(nbClusters_);
155
156 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
157 pValuesForTheCurrentYear[numSpace]
158 = new VCardType::IntermediateValuesDeepType[nbClusters_];
159
160 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
161 for (unsigned int i = 0; i != nbClusters_; ++i)
162 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
163
164 for (unsigned int i = 0; i != nbClusters_; ++i)
165 {
166 AncestorType::pResults[i].initializeFromStudy(*study);
167 AncestorType::pResults[i].reset();
168 }
169 }
170 else
171 {
172 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
173 {
174 pValuesForTheCurrentYear[numSpace] = nullptr;
175 }
176
177 AncestorType::pResults.clear();
178 }
179 // Next
180 NextType::initializeFromArea(study, area);
181 }
182
183 size_t getMaxNumberColumns() const

186 }

188 void yearBegin(unsigned int year, unsigned int numSpace)
189 {

195 // Next variable
196 NextType::yearBegin(year, numSpace);

199 void yearEnd(unsigned int year, unsigned int numSpace)
200 {
201 for (unsigned int clusterIndex = 0; clusterIndex < nbClusters_; ++clusterIndex)
202 {
203 // Compute all statistics from hourly results for the current year (daily, weekly, monthly, ...)
204 pValuesForTheCurrentYear[numSpace][clusterIndex].computeStatisticsForTheCurrentYear();
205 }
206 // Next variable
207 NextType::yearEnd(year, numSpace);
208 }

210 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
211 unsigned int nbYearsForCurrentSummary)
212 {
213 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
214 {
215 for (unsigned int clusterIndex = 0; clusterIndex < nbClusters_; ++clusterIndex)
216 {
217 // Merge all those values with the global results
218 AncestorType::pResults[clusterIndex].merge(numSpaceToYear[numSpace],
219 pValuesForTheCurrentYear[numSpace][clusterIndex]);
220 }
221 }
222
223 // Next variable
224 NextType::computeSummary(numSpaceToYear, nbYearsForCurrentSummary);
225 }

233 void hourForEachArea(State& state, unsigned int numSpace)
234 {
235 for (uint clusterIndex = 0; clusterIndex != state.area->shortTermStorage.count();
236 ++clusterIndex)
237 {
238 // ST storage injection for the current cluster and this hour
239 pValuesForTheCurrentYear[numSpace][clusterIndex].hour[state.hourInTheYear]
240 = state.hourlyResults->ShortTermStorage[state.hourInTheWeek].injection[clusterIndex];
241 }
242
243 // Next variable
244 NextType::hourForEachArea(state, numSpace);

269 void localBuildAnnualSurveyReport(SurveyResults& results,
270 int fileLevel,
271 int precision,
272 unsigned int numSpace) const
273 {
274 // Initializing external pointer on current variable non applicable status
275 results.isCurrentVarNA = AncestorType::isNonApplicable;
276
277 if (AncestorType::isPrinted[0])
278 {
279 assert(NULL != results.data.area);
280 const auto& shortTermStorage = results.data.area->shortTermStorage;
281
282 // Write the data for the current year
283 for (uint clusterIndex = 0; clusterIndex < nbClusters_; ++clusterIndex)
284 {
285 // Write the data for the current year
286 const auto* cluster = shortTermStorage.storagesByIndex[clusterIndex];
287 results.variableCaption = cluster->properties.name;
288 results.variableUnit = VCardType::Unit();
289 pValuesForTheCurrentYear[numSpace][clusterIndex]
290 .template buildAnnualSurveyReport<VCardType>(results, fileLevel, precision);
291 }
292 }
293 }
