122 template<int CDataLevel, int CFile>
123 struct Statistics
124 {
125 enum
126 {
127 count
128 = ((VCardType::categoryDataLevel & CDataLevel && VCardType::categoryFileLevel & CFile)
129 ? (NextType::template Statistics<CDataLevel, CFile>::count
130 + VCardType::columnCount * ResultsType::count)
131 : NextType::template Statistics<CDataLevel, CFile>::count),

135public:

141 void initializeFromStudy(Data::Study& study)
142 {
143 pNbYearsParallel = study.maxNbYearsInParallel;
144
145 // Average on all years
146 AncestorType::pResults.initializeFromStudy(study);
147 AncestorType::pResults.reset();
148
149 // Intermediate values
150 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
151 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
152 pValuesForTheCurrentYear[numSpace].initializeFromStudy(study);
153
154 // Next
155 NextType::initializeFromStudy(study);
156 }

168 }

170 void simulationBegin()

172 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)

204 }

206 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
207 unsigned int nbYearsForCurrentSummary)
208 {
209 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
210 {
211 // Merge all those values with the global results
212 AncestorType::pResults.merge(numSpaceToYear[numSpace],
213 pValuesForTheCurrentYear[numSpace]);
214 }
215
216 // Next variable
217 NextType::computeSummary(numSpaceToYear, nbYearsForCurrentSummary);
218 }

232 void hourForEachLink(State& state, unsigned int numSpace)
233 {
234 // Flow assessed over all MC years (linear)
235 pValuesForTheCurrentYear[numSpace].hour[state.hourInTheYear]
236 += Yuni::Math::Abs(state.ntc.ValeurDuFlux[state.link->index]);
237 // Next item in the list
238 NextType::hourForEachLink(state, numSpace);
239 }

268 results, fileLevel, precision);
269 }
