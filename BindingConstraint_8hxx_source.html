109template<class Env>
110inline std::string BindingConstraint::timeSeriesFileName(const Env &env) const {
111 switch (operatorType()) {
112 case BindingConstraint::opLess:
113 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_lt" + ".txt";
114 case BindingConstraint::opGreater:
115 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_gt" + ".txt";
116 case BindingConstraint::opEquality:
117 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_eq" + ".txt";
118 default:
119 logs.error("Cannot load/save time series of type other that eq/gt/lt");
120 return "";
121 }
122}
