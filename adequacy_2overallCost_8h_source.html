245 void hourForEachArea(State& state, unsigned int numSpace)
246 {
247 auto area = state.area;
248 auto& thermal = state.thermal;
249 // Total UnsupliedEnergy emissions
250 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] +=
251 // Current Hydro Storage generation
252 (state.hourlyResults->ValeursHorairesDeDefaillancePositive[state.hourInTheWeek]
253 * area->thermal.unsuppliedEnergyCost)
254 + ((state.hourlyResults->ValeursHorairesDeDefaillanceNegative[state.hourInTheWeek]
255 + state.resSpilled.entry[area->index][state.hourInTheWeek])
256 * area->thermal.spilledEnergyCost);
257
258 // Hydro costs : water value and pumping
259 pValuesForTheCurrentYear[numSpace].hour[state.hourInTheYear]
260 += state.problemeHebdo->CaracteristiquesHydrauliques[state.area->index]
261 .WeeklyWaterValueStateRegular
262 * (state.hourlyResults->TurbinageHoraire[state.hourInTheWeek]
263 - area->hydro.pumpingEfficiency
264 * state.hourlyResults->PompageHoraire[state.hourInTheWeek]);
265
266 // Thermal costs
267 for (uint clusterIndex = 0; clusterIndex != area->thermal.clusterCount(); ++clusterIndex)
268 {
269 pValuesForTheCurrentYear[numSpace][state.hourInTheYear]
270 += thermal[area->index].thermalClustersOperatingCost[clusterIndex];
271 }
272
273 // Next variable
274 NextType::hourForEachArea(state, numSpace);
275 }
