210 void initializeFromStudy(Data::Study& study)
211 {
212 typedef
213 typename VCardType::VCardOrigin::IntermediateValuesBaseType IntermediateValuesBaseType;
214 pNbYearsParallel = study.maxNbYearsInParallel;
215
216 // Intermediate values
217 VarT<Container::EndOfList>::InitializeResultsFromStudy(AncestorType::pResults, study);
218 pValuesForTheCurrentYear = new IntermediateValuesBaseType[pNbYearsParallel];
219 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
220 {
221 VariableAccessorType::InitializeAndReset(pValuesForTheCurrentYear[numSpace], study);
222 }
223
224 auto& limits = study.runtime.rangeLimits;
225
226 pRatioYear = 100. / (double)limits.year[Data::rangeCount];
227 pRatioDay = 100. / (double)limits.day[Data::rangeCount];
228 pRatioMonth = 100. / (double)limits.month[Data::rangeCount];
229 pRatioWeek = 100. / (double)limits.week[Data::rangeCount];
230
231 // Next
232 NextType::initializeFromStudy(study);
233 }

396 template<class V, class SetT>
397 void internalSpatialAggregate(V& allVars, uint year, const SetT& set)
398 {
399 typedef typename VCardType::VCardOrigin VCardOrigin;
400 // Reset the results
401 VariableAccessorType::Reset(pValuesForTheCurrentYear[0]);
402
403 // Make the spatial cluster
404 if (!set.empty())
405 {
406 auto end = set.end();
407 for (auto i = set.begin(); i != end; ++i)
408 {
409 allVars.template computeSpatialAggregateWith<
410 typename VCardType::VCardForSpatialAggregate> //<typename VCardType::VCardOrigin>
411 (pValuesForTheCurrentYear[0], *i /* the current area */, 0);
412 }
413
414 // The spatial cluster may be an average
415 if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateAverage)
416 {
417 VariableAccessorType::MultiplyHourlyResultsBy(pValuesForTheCurrentYear[0],
418 1. / set.size());
419 }
420 // The spatial cluster may be an average
421 if (VCardType::VCardOrigin::spatialAggregate
422 & Category::spatialAggregateSumThen1IfPositive)
423 {
424 VariableAccessorType::SetTo1IfPositive(pValuesForTheCurrentYear[0]);
425 }
426 if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateOr)
427 {
428 VariableAccessorType::Or(pValuesForTheCurrentYear[0]);
429 }
430 }
431 else
432 {
433 assert(!set.empty() && "The set should not be empty at this point");
434 }
435
436 // Compute all statistics for the current year (daily,weekly,monthly,...)
437 VariableAccessorType::template ComputeStatistics<VCardOrigin>(pValuesForTheCurrentYear[0]);
438 VariableAccessorType::ComputeSummary(pValuesForTheCurrentYear[0],
439 AncestorType::pResults,
440 year);
441 }

443 template<class V, class SetT>
444 void internalSpatialAggregateForCurrentYear(V& allVars, const SetT& set, uint numSpace)
445 {
446 typedef typename VCardType::VCardOrigin VCardOrigin;
447 // Reset the results
448 VariableAccessorType::Reset(pValuesForTheCurrentYear[numSpace]);
449
450 // Make the spatial cluster
451 if (!set.empty())
452 {
453 auto end = set.end();
454 for (auto i = set.begin(); i != end; ++i)
455 {
456 allVars.template computeSpatialAggregateWith<
457 typename VCardType::VCardForSpatialAggregate> //<typename VCardType::VCardOrigin>
458 (pValuesForTheCurrentYear[numSpace], *i /* the current area */, numSpace);
459 }
460
461 // The spatial cluster may be an average
462 if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateAverage)
463 {
464 VariableAccessorType::MultiplyHourlyResultsBy(pValuesForTheCurrentYear[numSpace],
465 1. / set.size());
466 }
467 // The spatial cluster may be an average
468 if (VCardType::VCardOrigin::spatialAggregate
469 & Category::spatialAggregateSumThen1IfPositive)
470 {
471 VariableAccessorType::SetTo1IfPositive(pValuesForTheCurrentYear[numSpace]);
472 }
473 if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateOr)
474 {
475 VariableAccessorType::Or(pValuesForTheCurrentYear[numSpace]);
476 }
477 }
478 else
479 {
480 assert(!set.empty() && "The set should not be empty at this point");
481 }
482
483 // Compute all statistics for the current year (daily,weekly,monthly,...)
484 VariableAccessorType::template ComputeStatistics<VCardOrigin>(
485 pValuesForTheCurrentYear[numSpace]);
486 }
