97 template<class S, class VCardT>
98 void buildSurveyReport(SurveyResults& report,
99 const S& results,
100 int dataLevel,
101 int fileLevel,
102 int precision) const
103 {
104 if (!(fileLevel & Category::id))
105 {
106 switch (precision)
107 {
108 case Category::hourly:
109 InternalExportValues<maxHoursInAYear, VCardT, Category::hourly>(
110 report, Memory::RawPointer(avgdata.hourly));
111 break;
112 case Category::daily:
113 InternalExportValues<maxDaysInAYear, VCardT, Category::daily>(report,

115 break;
116 case Category::weekly:
117 InternalExportValues<maxWeeksInAYear, VCardT, Category::weekly>(report,
118 avgdata.weekly);
119 break;
120 case Category::monthly:

122 avgdata.monthly);
123 break;
124 case Category::annual:
125 InternalExportValues<1, VCardT, Category::annual>(report, avgdata.year);

127 }
128 }

134 template<class VCardT>
135 void buildDigest(SurveyResults& report, int digestLevel, int dataLevel) const
136 {
137 const bool isCluster = (VCardT::categoryFileLevel & Category::de)
138 || (VCardT::categoryFileLevel & Category::de_res);
139 const bool isBindingConstraint = VCardT::categoryFileLevel & Category::bc;
140 const bool isDigest = digestLevel & Category::digestAllYears;
141 if ((dataLevel & Category::area || dataLevel & Category::setOfAreas) && isDigest
142 && !isCluster && !isBindingConstraint)
143 {
144 assert(report.data.columnIndex < report.maxVariables && "Column index out of bounds");
145
146 report.captions[0][report.data.columnIndex] = report.variableCaption;
147 report.captions[1][report.data.columnIndex] = report.variableUnit;
148 report.captions[2][report.data.columnIndex]
149 = (report.variableCaption == "LOLP") ? "values" : "EXP";
150
151 // Precision
152 report.precision[report.data.columnIndex]
153 = PrecisionToPrintfFormat<VCardT::decimal>::Value();
154 // Value
155 report.values[report.data.columnIndex][report.data.rowIndex] = avgdata.allYears;
156 // Non applicability
157 report.digestNonApplicableStatus[report.data.rowIndex][report.data.columnIndex]
158 = *report.isCurrentVarNA;
159
160 ++(report.data.columnIndex);
161 }
162 // Next
163 NextType::template buildDigest<VCardT>(report, digestLevel, dataLevel);
164 }

183 template<uint Size, class VCardT, int PrecisionT>
184 void InternalExportValues(SurveyResults& report, const double* array) const
185 {
186 assert(array);
187 assert(report.data.columnIndex < report.maxVariables && "Column index out of bounds");
188
189 // Caption
190 report.captions[0][report.data.columnIndex] = report.variableCaption;
191 report.captions[1][report.data.columnIndex] = report.variableUnit;
192 report.captions[2][report.data.columnIndex]
193 = (report.variableCaption == "LOLP") ? "values" : "EXP";
194 // Precision
195 report.precision[report.data.columnIndex]
196 = PrecisionToPrintfFormat<VCardT::decimal>::Value();
197 // Non applicability
198 report.nonApplicableStatus[report.data.columnIndex] = *report.isCurrentVarNA;
199
200 // Values
201 switch (PrecisionT)
202 {
203 case Category::annual:
204 {
205 double& target = *(report.values[report.data.columnIndex]);
206 target = 0;
207 for (uint i = 0; i != avgdata.nbYearsCapacity; ++i)
208 target += array[i];
209 avgdata.allYears = target;
210 break;
211 }
212 default:
213 (void)::memcpy(report.values[report.data.columnIndex], array, sizeof(double) * Size);
214 break;
215 }
216
217 // Next column index
218 ++report.data.columnIndex;
219 }
