243template<class NextT>
244void List<NextT>::buildSurveyReport(SurveyResults& results,
245 int dataLevel,
246 int fileLevel,
247 int precision) const
248{
249 // Reset
250 results.data.columnIndex = 0;
251
252 // The new filename
253 results.data.filename.clear();
254 results.data.filename << results.data.output << SEP;
255 Category::FileLevelToStream(results.data.filename, fileLevel);
256 results.data.filename << '-';
257 Category::PrecisionLevelToStream(results.data.filename, precision);
258 results.data.filename << ".txt";
259
260 // Ask to all variables
261 NextType::buildSurveyReport(results, dataLevel, fileLevel, precision);
262
263 // If the column index is still equals to 0, that would mean we have nothing
264 // to do (there is no data to write)
265 if (results.data.columnIndex > 0)
266 results.saveToFile(dataLevel, fileLevel, precision);
267}

310template<class NextT>
311void List<NextT>::exportSurveyResults(bool global,
312 const Yuni::String& output,
313 unsigned int numSpace,
314 IResultWriter& writer)
315{
316 using namespace Antares;
317
318 // Infos
319 if (global)
320 {
321 logs.info(); // empty line
322 logs.checkpoint() << "Exporting the survey results...";
323 }
324 else
325 logs.info() << "Exporting the annual results";
326
327 auto survey = std::make_shared<SurveyResults>(*pStudy, output, writer);
328
329 // Year by year ?
330 survey->yearByYearResults = !global;
331
332 if (global)
333 {
334 // alias to the type of the report builder
335 using Builder = SurveyReportBuilder<true, ListType>;
336 // Building the survey results for each possible state
337 Builder::Run(*this, *survey);
338
339 // Exporting the Grid (information about the study)
340 survey->exportGridInfos();
341
342 // Exporting the digest
343 // The digest must be exported after the real report because some values
344 // are computed at this moment.
345 Builder::RunDigest(*this, *survey, writer);
346 }
347 else
348 {
349 // alias to the type of the report builder
350 using Builder = SurveyReportBuilder<false, ListType>;
351 // Building the survey results for each possible state
352 Builder::Run(*this, *survey, numSpace);
353 }
354}
