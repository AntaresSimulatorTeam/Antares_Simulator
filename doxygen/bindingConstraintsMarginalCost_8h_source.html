86/*
87 Marginal cost associated to binding constraints :
88 Suppose that the BC is hourly,
89 - if binding constraint is not saturated (rhs is not reached) for a given hour, the value is 0;
90 - if binding constraint is saturated (rhs is reached), the value is the total benefit (�/MW) for
91 the system that would result in increasing the BC's rhs of 1 MW.
92*/

229 void weekBegin(State& state)
230 {
231 if (isInitialized())
232 {
233 auto numSpace = state.numSpace;
234 // For daily binding constraints, getting daily marginal price
235 using namespace Data;
236 switch (associatedBC_->type())
237 {
238 case BindingConstraint::typeHourly:
239 case BindingConstraint::typeUnknown:
240 case BindingConstraint::typeMax:
241 break;
242
243 case BindingConstraint::typeDaily:
244 {
245 int dayInTheYear = state.weekInTheYear * 7;
246 for (int dayInTheWeek = 0; dayInTheWeek < 7; dayInTheWeek++)
247 {
248 pValuesForTheCurrentYear[numSpace].day[dayInTheYear]
249 -= state.problemeHebdo
250 ->ResultatsContraintesCouplantes[associatedBC_][dayInTheWeek];
251
252 dayInTheYear++;
253 }
254 break;
255 }
256
257 // For weekly binding constraints, getting weekly marginal price
258 case BindingConstraint::typeWeekly:
259 {
260 uint weekInTheYear = state.weekInTheYear;
261 double weeklyValue = -state.problemeHebdo
262 ->ResultatsContraintesCouplantes[associatedBC_][0];
263
264 pValuesForTheCurrentYear[numSpace].week[weekInTheYear] = weeklyValue;
265
266 int dayInTheYear = state.weekInTheYear * 7;
267 for (int dayInTheWeek = 0; dayInTheWeek < 7; dayInTheWeek++)
268 {
269 pValuesForTheCurrentYear[numSpace].day[dayInTheYear] = weeklyValue;
270 dayInTheYear++;
271 }
272 break;
273 }
274 }
275 }
276 NextType::weekBegin(state);
277 }

309 void localBuildAnnualSurveyReport(
310 SurveyResults& results,

333 void buildSurveyReport(SurveyResults& results,

337 {

339 // ------------------------------

343 if ((dataLevel & VCardType::categoryDataLevel) && (fileLevel & VCardType::categoryFileLevel)
344 && (precision & VCardType::precision))
345 {
346 results.isPrinted = AncestorType::isPrinted;
347 results.isCurrentVarNA[0] = isCurrentOutputNonApplicable(precision);
348 results.variableCaption = getBindConstraintCaption();

350 VariableAccessorType::template BuildSurveyReport<VCardType>(results,
351 AncestorType::pResults,
352 dataLevel,
353 fileLevel,
354 precision,
355 false);
356 }
357 }
358 NextType::buildSurveyReport(results, dataLevel, fileLevel, precision);

376 bool isCurrentOutputNonApplicable(int precision) const
377 {
378 using namespace Antares::Data;
379 // The current marginal prices to print becomes non applicable if they have a precision
380 // (hour, day, week, ...) smaller than the associated binding constraint granularity.
381 // Ex : if the BC is daily and we try to print hourly associated marginal prices,
382 // then these prices are set to N/A
383 switch (associatedBC_->type())
384 {
385 case BindingConstraint::typeUnknown:
386 case BindingConstraint::typeMax:
387 return true;
388 default:
389 const auto precision_bc = 1 << (associatedBC_->type() - 1);
390 return precision < precision_bc;
391 }
392 }
