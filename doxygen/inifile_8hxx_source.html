59template<class U>
60IniFile::Property* IniFile::Section::add(const AnyString& key, const U& value)
61{
62 auto* p = new Property(key, value);
63 if (!lastProperty)
64 firstProperty = p;
65 else
66 lastProperty->next = p;
67 lastProperty = p;
68 return p;
69}
