227 void hourForEachLink(State& state, unsigned int numSpace)
228 {
229 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] = std::abs(
230 state.problemeHebdo->VariablesDualesDesContraintesDeNTC[state.hourInTheWeek]
231 .VariableDualeParInterconnexion[state.link->index]);
232
233 // This value should be reset to zero if (flow_lowerbound) < flow < (flow_upperbound) (with
234 // signed values)
235 double flow = state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
236 .ValeurDuFlux[state.link->index];
237 double flow_lowerbound = -state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
238 .ValeurDeNTCExtremiteVersOrigine[state.link->index];
239 double flow_upperbound = state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
240 .ValeurDeNTCOrigineVersExtremite[state.link->index];
241
242 if (flow - 0.001 > flow_lowerbound && flow + 0.001 < flow_upperbound)
243 {
244 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] = 0.;
245 }
246
247 // Next item in the list
248 NextType::hourForEachLink(state, numSpace);
249 }
