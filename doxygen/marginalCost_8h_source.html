223 void hourForEachLink(State& state, unsigned int numSpace)
224 {
225 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] = Yuni::Math::Abs(
226 state.problemeHebdo->VariablesDualesDesContraintesDeNTC[state.hourInTheWeek]
227 .VariableDualeParInterconnexion[state.link->index]);
228
229 // This value should be reset to zero if (flow_lowerbound) < flow < (flow_upperbound) (with
230 // signed values)
231 double flow
232 = state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek].ValeurDuFlux[state.link->index];
233 double flow_lowerbound = -state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
234 .ValeurDeNTCExtremiteVersOrigine[state.link->index];
235 double flow_upperbound = state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
236 .ValeurDeNTCOrigineVersExtremite[state.link->index];
237
238 if (flow - 0.001 > flow_lowerbound && flow + 0.001 < flow_upperbound)
239 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] = 0.;
240
241 // Next item in the list
242 NextType::hourForEachLink(state, numSpace);
243 }
