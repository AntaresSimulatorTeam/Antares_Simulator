228 void hourForEachLink(State& state, unsigned int numSpace)
229 {
230#define UPSTREAM_PRICE \
231 state.problemeHebdo->ResultatsHoraires[state.link->from->index] \
232 .CoutsMarginauxHoraires[state.hourInTheWeek]
233#define DOWNSTREAM_PRICE \
234 state.problemeHebdo->ResultatsHoraires[state.link->with->index] \
235 .CoutsMarginauxHoraires[state.hourInTheWeek]
236
237 // Congestion Fee
238 pValuesForTheCurrentYear[numSpace].hour[state.hourInTheYear]
239 = state.ntc.ValeurDuFlux[state.link->index] * (UPSTREAM_PRICE - DOWNSTREAM_PRICE);
240 // Next item in the list
241 NextType::hourForEachLink(state, numSpace);
242
243#undef UPSTREAM_PRICE
244#undef DOWNSTREAM_PRICE
245 }
