110template<class T, class ReadWriteT, uint PrecisionT>
111bool Matrix<T, ReadWriteT, PrecisionT>::cellValue(int x, int y, const Yuni::String& value)
112{
113 if (pMatrix && (uint)x < pMatrix->width && (uint)y < pMatrix->height)
114 {
115 double v;
116 if (value.to(v))
117 {
118 // The value must be rounded if not a decimal value
119 if (Yuni::Static::Type::IsDecimal<ReadWriteT>::No)
120 v = Yuni::Math::Round(v);
121 else
122 v = Yuni::Math::Round(v, matrixPrecision);
123
124 if (!Yuni::Math::Equals((double)pMatrix->entry[x][y], v))
125 {
126 pMatrix->entry[x][y] = static_cast<T>(static_cast<ReadWriteT>(v));
127 pMatrix->markAsModified();
128 }
129 return true;
130 }
131 }
132 return false;
133}

142template<class T, class ReadWriteT, uint PrecisionT>
143bool Matrix<T, ReadWriteT, PrecisionT>::ensureDataAreLoaded()
144{
145 using namespace Yuni;
146
147 GUIBeginUpdate();
148
149 if (!(!study) && pMatrix && pMatrix->jit)
150 {
151 auto& jit = *pMatrix->jit;
152 // Timestamp for last modification
153 auto timestamp = IO::File::LastModificationTime(jit.sourceFilename);
154 if (jit.lastModification != timestamp)
155 {
156 jit.lastModification = timestamp;
157 if (!jit.modified)
158 jit.alreadyLoaded = false;
159 }
160 }
161
162 if (!(!study) && pMatrix && (!JIT::IsReady(pMatrix->jit) || invalidate))
163 {
164 if (!pMatrix->jit)
165 {
166 GUIEndUpdate();
167 return true;
168 }
169
170 auto& jit = *pMatrix->jit;
171 ::wxBeginBusyCursor();
172
173 auto& mainFrm = *Forms::ApplWnd::Instance();
174
175 // Working...
176 WIP::Locker wip;
177 GUILocker locker;
178
179 // The following instructions seem to seriously slow down
180 // the execution of the program :
181 // # wxBusyInfo info(wxT("Loading data..."), Forms::ApplWnd::Instance());
182 // # wxTheApp->Yield();
183
184 if (not JIT::IsReady(pMatrix->jit))
185 {
186 // We can call SetStatusText from here, it may produce some strange segv
187 // on Linux. Nothing indicates that it has the same issue on Windows.
188 // mainFrm.SetStatusText(wxString(wxT(" Loading "))
189 // << wxStringFromUTF8(jit.sourceFilename) << wxT("..."));
190 logs.info() << "loading '" << jit.sourceFilename << "'";
191 }
192
193 if (not pMatrix->forceReload(true))
194 {
195 if (pMatrix->jit)
196 logs.error() << "I/O Error: Impossible to load '" << jit.sourceFilename << "'";
197 else
198 logs.error() << "I/O Error: Impossible to load the data";
199 }
200 // Notify
201 onMatrixLoad();
202
203 mainFrm.resetDefaultStatusBarText();
204
205 // restore cursor
206 ::wxEndBusyCursor();
207 GUIEndUpdate();
208 return true;
209 }
210
211 GUIEndUpdate();
212 return false;
213}

249template<class T, class ReadWriteT, uint PrecisionT>
250bool Matrix<T, ReadWriteT, PrecisionT>::circularShiftRowsUntilDate(MonthName month, uint daymonth)
251{
252 if (!study || !pMatrix)
253 return false;
254 if (daymonth > 30)
255 {
256#ifndef NDEBUG
257 logs.warning() << "invalid internal parameter: daymonth is bteween 0..30";
258#endif
259 return false;
260 }
261
262 auto& calendar = study->calendar;
263 uint relativeMonth = calendar.mapping.months[(uint)month];
264 uint daysForMonth = calendar.months[relativeMonth].days;
265 if (daymonth >= daysForMonth)
266 return false;
267
268 uint dayYear = calendar.months[relativeMonth].daysYear.first + daymonth;
269 if (dayYear >= calendar.maxDaysInYear)
270 return false;
271 uint hourindex = calendar.days[dayYear].hours.first;
272
273 switch (this->precision())
274 {
275 case Date::hourly:
276 {
277 GUIBeginUpdate();
278 pMatrix->circularShiftRows(hourindex);
279 MarkTheStudyAsModified(study);
280 GUIEndUpdate();
281 return true;
282 }
283 case Date::daily:
284 {
285 GUIBeginUpdate();
286 pMatrix->circularShiftRows(dayYear);
287 MarkTheStudyAsModified(study);
288 GUIEndUpdate();
289 return true;
290 }
291 case Date::monthly:
292 {
293 GUIBeginUpdate();
294 pMatrix->circularShiftRows(month);
295 MarkTheStudyAsModified(study);
296 GUIEndUpdate();
297 return true;
298 }
299 // case Date::stepSecond:
300 // case Date::stepMinute:
301 case Date::weekly:
302 case Date::annual:
303 case Date::stepNone:
304 case Date::stepAny:
305 logs.error() << "row shifting not allowed here";
306 break;
307 }
308
309 return false;
310}
