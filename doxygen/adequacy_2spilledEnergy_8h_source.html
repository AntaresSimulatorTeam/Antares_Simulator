99template<class NextT = Container::EndOfList>
100class SpilledEnergy: public Variable::IVariable<SpilledEnergy<NextT>, NextT, VCardSpilledEnergy>

102public:

121 template<int CDataLevel, int CFile>
122 struct Statistics
123 {
124 enum
125 {
126 count = ((VCardType::categoryDataLevel & CDataLevel
127 && VCardType::categoryFileLevel & CFile)
128 ? (NextType::template Statistics<CDataLevel, CFile>::count

134public:

140 void initializeFromStudy(Data::Study& study)
141 {
142 pNbYearsParallel = study.maxNbYearsInParallel;
143
144 // Intermediate values
145 InitializeResultsFromStudy(AncestorType::pResults, study);
146
147 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
148 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
149 {
150 pValuesForTheCurrentYear[numSpace].initializeFromStudy(study);
151 }
152
153 // Next
154 NextType::initializeFromStudy(study);
155 }

175 void simulationBegin()
176 {
177 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)

182 NextType::simulationBegin();

187 NextType::simulationEnd();

198 void yearEndBuild(State& state, unsigned int year, unsigned int numSpace)

204 void yearEnd(unsigned int year, unsigned int numSpace)

208
209 // Next variable
210 NextType::yearEnd(year, numSpace);
211 }
212
213 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
214 unsigned int nbYearsForCurrentSummary)
215 {
216 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
217 {
218 // Merge all those values with the global results
219 AncestorType::pResults.merge(numSpaceToYear[numSpace] /*year*/,
220 pValuesForTheCurrentYear[numSpace]);
221 }
222
223 // Next variable
224 NextType::computeSummary(numSpaceToYear, nbYearsForCurrentSummary);
225 }

227 void hourBegin(unsigned int hourInTheYear)
228 {
229 // Next variable

231 }
232
233 void hourForEachArea(State& state, unsigned int numSpace)
234 {
235 assert(state.hourlyResults && "Invalid pointer to simplex results");
236
237 // Total SpilledEnergy emissions
238 pValuesForTheCurrentYear[numSpace][state.hourInTheYear]
239 = +state.hourlyResults->ValeursHorairesDeDefaillanceNegative[state.hourInTheWeek]
240 + state.resSpilled.entry[state.area->index][state.hourInTheWeek];
241
242 // Next variable
243 NextType::hourForEachArea(state, numSpace);
244 }

253 void localBuildAnnualSurveyReport(SurveyResults& results,
254 int fileLevel,

259 results.isCurrentVarNA = AncestorType::isNonApplicable;
260
261 if (AncestorType::isPrinted[0])
262 {
263 // Write the data for the current year
264 results.variableCaption = VCardType::Caption();
265 results.variableUnit = VCardType::Unit();
266 pValuesForTheCurrentYear[numSpace]
267 .template buildAnnualSurveyReport<VCardType>(results, fileLevel, precision);
268 }
269 }
