101template<class Env>
102inline std::string BindingConstraint::timeSeriesFileName(const Env &env) const {
103 switch (operatorType()) {
104 case BindingConstraint::opLess:
105 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_lt" + ".txt";
106 case BindingConstraint::opGreater:
107 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_gt" + ".txt";
108 case BindingConstraint::opEquality:
109 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_eq" + ".txt";
110 default:
111 logs.error("Cannot load/save time series of type other that eq/gt/lt");
112 return "";
113 }
114}
