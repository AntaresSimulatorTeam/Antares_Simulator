108template<class Env>
109inline std::string BindingConstraint::timeSeriesFileName(const Env& env) const
110{
111 switch (operatorType())
112 {
113 case BindingConstraint::opLess:
114 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_lt"
115 + ".txt";
116 case BindingConstraint::opGreater:
117 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_gt"
118 + ".txt";
119 case BindingConstraint::opEquality:
120 return std::string() + env.folder.c_str() + Yuni::IO::Separator + id().c_str() + "_eq"
121 + ".txt";
122 default:
123 logs.error("Cannot load/save time series of type other that eq/gt/lt");
124 return "";
125 }
126}
