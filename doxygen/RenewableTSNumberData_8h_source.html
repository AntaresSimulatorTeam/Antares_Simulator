58inline uint renewableTSNumberData::get(const Antares::Data::RenewableCluster* cluster,
59 const uint year) const
60{
61 assert(cluster != nullptr);
62 if (year < pTSNumberRules.height && cluster->areaWideIndex < pTSNumberRules.width)
63 {
64 const uint index = cluster->areaWideIndex;
65 return pTSNumberRules[index][year];
66 }
67 return 0;
68}
