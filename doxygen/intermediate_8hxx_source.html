61template<class VCardT>
62inline void IntermediateValues::buildAnnualSurveyReport(SurveyResults& report,
63 int fileLevel,
64 int precision) const
65{
66 if (!(fileLevel & Category::FileLevel::id))
67 {
68 switch (precision)
69 {
70 case Category::hourly:
71 internalExportAnnualValues<HOURS_PER_YEAR, VCardT>(report, hour, false);
72 break;
73 case Category::daily:
74 internalExportAnnualValues<DAYS_PER_YEAR, VCardT>(report, day, false);
75 break;
76 case Category::weekly:
77 internalExportAnnualValues<WEEKS_PER_YEAR, VCardT>(report, week, false);
78 break;
79 case Category::monthly:
80 internalExportAnnualValues<MONTHS_PER_YEAR, VCardT>(report, month, false);
81 break;
82 case Category::annual:
83 internalExportAnnualValues<1, VCardT>(report, &year, true);
84 break;
85 }
86 }
87}

89template<unsigned int Size, class VCardT, class A>
90void IntermediateValues::internalExportAnnualValues(SurveyResults& report,
91 const A& array,
92 bool annual) const
93{
94 using namespace Yuni;
95 assert(report.data.columnIndex < report.maxVariables && "Column index out of bounds");
96
97 // Caption
98 report.captions[0][report.data.columnIndex] = report.variableCaption;
99 report.captions[1][report.data.columnIndex] = report.variableUnit;
100 report.captions[2][report.data.columnIndex] = nullptr;
101 // Precision
102 report.precision[report.data.columnIndex] = PrecisionToPrintfFormat<VCardT::decimal>::Value();
103 // Non applicability
104 report.nonApplicableStatus[report.data.columnIndex] = *report.isCurrentVarNA;
105
106 // Values
107 if (not annual)
108 {
109 (void)::memcpy(report.values[report.data.columnIndex], array, sizeof(double) * Size);
110 }
111 else
112 {
113 double& target = *(report.values[report.data.columnIndex]);
114 target = year;
115 }
116
117 // Next column index
118 ++report.data.columnIndex;
119}
