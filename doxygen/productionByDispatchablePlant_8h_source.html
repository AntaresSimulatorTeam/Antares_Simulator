160 void initializeFromArea(Data::Study* study, Data::Area* area)
161 {
162 // Get the number of years in parallel
163 pNbYearsParallel = study->maxNbYearsInParallel;
164 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
165 pminOfTheClusterForYear = new double*[pNbYearsParallel];
166
167 // Get the area
168 pSize = area->thermal.list.enabledCount();
169 if (pSize)
170 {
171 AncestorType::pResults.resize(pSize);
172
173 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
174 {
175 pValuesForTheCurrentYear[numSpace] = new VCardType::IntermediateValuesDeepType
176 [pSize];
177 }
178
179 // Minimum power values of the cluster for the whole year - from the solver in the
180 // accurate mode not to be displayed in the output \todo think of a better place like
181 // the DispatchableMarginForAllAreas done at the beginning of the year
182
183 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
184 {
185 pminOfTheClusterForYear[numSpace] = new double[pSize * HOURS_PER_YEAR];
186 }
187
188 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
189 {
190 for (unsigned int i = 0; i != pSize; ++i)
191 {
192 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
193 }
194 }
195
196 for (unsigned int i = 0; i != pSize; ++i)
197 {
198 AncestorType::pResults[i].initializeFromStudy(*study);
199 AncestorType::pResults[i].reset();
200 }
201 }
202 else
203 {
204 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
205 {
206 pValuesForTheCurrentYear[numSpace] = nullptr;
207 pminOfTheClusterForYear[numSpace] = nullptr;
208 }
209
210 AncestorType::pResults.clear();
211 }
212 // Next
213 NextType::initializeFromArea(study, area);
214 }

254 void yearEndBuildPrepareDataForEachThermalCluster(State& state,
255 uint year,
256 unsigned int numSpace)
257 {
258 for (unsigned int i = 0; i <= state.study.runtime.rangeLimits.hour[Data::rangeEnd]; ++i)
259 {
260 state.thermalClusterProductionForYear[i] += pValuesForTheCurrentYear
261 [numSpace]
262 [state.thermalCluster->areaWideIndex]
263 .hour[i];
264 state.thermalClusterPMinOfTheClusterForYear[i] += pminOfTheClusterForYear
265 [numSpace][(state.thermalCluster->areaWideIndex * HOURS_PER_YEAR) + i];
266 }
267
268 // Next variable
269 NextType::yearEndBuildPrepareDataForEachThermalCluster(state, year, numSpace);

315 void hourForEachArea(State& state, unsigned int numSpace)
316 {
317 auto& area = state.area;
318 auto& thermal = state.thermal;
319 for (auto& cluster: area->thermal.list.each_enabled())
320 {
321 // Production for this hour
322 pValuesForTheCurrentYear[numSpace][cluster->areaWideIndex].hour[state.hourInTheYear]
323 += thermal[area->index].thermalClustersProductions[cluster->areaWideIndex];
324
325 pminOfTheClusterForYear[numSpace]
326 [(cluster->areaWideIndex * HOURS_PER_YEAR) + state.hourInTheYear]
327 = thermal[area->index].PMinOfClusters[cluster->areaWideIndex];
328 }
329
330 // Next variable
331 NextType::hourForEachArea(state, numSpace);
332 }
