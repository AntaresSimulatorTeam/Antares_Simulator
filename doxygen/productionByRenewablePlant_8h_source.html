152 void initializeFromArea(Data::Study* study, Data::Area* area)
153 {
154 // Get the number of years in parallel
155 pNbYearsParallel = study->maxNbYearsInParallel;
156 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
157
158 // Get the area
159 pSize = area->renewable.list.enabledCount();
160 if (pSize)
161 {
162 AncestorType::pResults.resize(pSize);
163
164 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
165 {
166 pValuesForTheCurrentYear[numSpace] = new VCardType::IntermediateValuesDeepType
167 [pSize];
168 }
169
170 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
171 {
172 for (unsigned int i = 0; i != pSize; ++i)
173 {
174 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
175 }
176 }
177
178 for (unsigned int i = 0; i != pSize; ++i)
179 {
180 AncestorType::pResults[i].initializeFromStudy(*study);
181 AncestorType::pResults[i].reset();
182 }
183 }
184 else
185 {
186 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
187 {
188 pValuesForTheCurrentYear[numSpace] = nullptr;
189 }
190
191 AncestorType::pResults.clear();
192 }
193 // Next
194 NextType::initializeFromArea(study, area);
195 }

286 state.year,
