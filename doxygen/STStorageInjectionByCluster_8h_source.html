138 void initializeFromArea(Data::Study* study, Data::Area* area)
139 {
140 // Get the number of years in parallel
141 pNbYearsParallel = study->maxNbYearsInParallel;
142 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
143
144 // Get the area
145 nbClusters_ = area->shortTermStorage.count();
146 if (nbClusters_)
147 {
148 AncestorType::pResults.resize(nbClusters_);
149
150 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
151 pValuesForTheCurrentYear[numSpace]
152 = new VCardType::IntermediateValuesDeepType[nbClusters_];
153
154 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
155 for (unsigned int i = 0; i != nbClusters_; ++i)
156 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
157
158 for (unsigned int i = 0; i != nbClusters_; ++i)
159 {
160 AncestorType::pResults[i].initializeFromStudy(*study);
161 AncestorType::pResults[i].reset();
162 }
163 }
164 else
165 {
166 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
167 {
168 pValuesForTheCurrentYear[numSpace] = nullptr;
169 }
170
171 AncestorType::pResults.clear();
172 }
173 // Next
174 NextType::initializeFromArea(study, area);
175 }

204 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
205 unsigned int nbYearsForCurrentSummary)
206 {
207 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
208 {
209 for (unsigned int clusterIndex = 0; clusterIndex < nbClusters_; ++clusterIndex)
210 {
211 // Merge all those values with the global results

232 // ST storage injection for the current cluster and this hour

263 void localBuildAnnualSurveyReport(SurveyResults& results,
264 int fileLevel,
265 int precision,
266 unsigned int numSpace) const
267 {
268 // Initializing external pointer on current variable non applicable status
269 results.isCurrentVarNA = AncestorType::isNonApplicable;
270
271 if (AncestorType::isPrinted[0])
272 {
273 assert(NULL != results.data.area);
274 const auto& shortTermStorage = results.data.area->shortTermStorage;
275
276 // Write the data for the current year
277 uint clusterIndex = 0;
278 for (const auto& cluster : shortTermStorage.storagesByIndex)
279 {
280 // Write the data for the current year
281 results.variableCaption = cluster.properties.name;
282 results.variableUnit = VCardType::Unit();
283 pValuesForTheCurrentYear[numSpace][clusterIndex]
284 .template buildAnnualSurveyReport<VCardType>(results, fileLevel, precision);
285
286 clusterIndex++;
287 }
288 }
289 }
