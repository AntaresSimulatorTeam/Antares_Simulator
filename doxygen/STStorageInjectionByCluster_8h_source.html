138 void initializeFromArea(Data::Study* study, Data::Area* area)
139 {
140 // Get the number of years in parallel
141 pNbYearsParallel = study->maxNbYearsInParallel;
142 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
143
144 // Get the area
145 nbClusters_ = area->shortTermStorage.count();
146 if (nbClusters_)
147 {
148 AncestorType::pResults.resize(nbClusters_);
149
150 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
151 {
152 pValuesForTheCurrentYear[numSpace] = new VCardType::IntermediateValuesDeepType
153 [nbClusters_];
154 }
155
156 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
157 {
158 for (unsigned int i = 0; i != nbClusters_; ++i)
159 {
160 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
161 }
162 }
163
164 for (unsigned int i = 0; i != nbClusters_; ++i)
165 {
166 AncestorType::pResults[i].initializeFromStudy(*study);
167 AncestorType::pResults[i].reset();
168 }
169 }
170 else
171 {
172 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
173 {
174 pValuesForTheCurrentYear[numSpace] = nullptr;
175 }
176
177 AncestorType::pResults.clear();
178 }
179 // Next
180 NextType::initializeFromArea(study, area);
181 }

271 void localBuildAnnualSurveyReport(SurveyResults& results,
272 int fileLevel,
273 int precision,

275 {
276 // Initializing external pointer on current variable non applicable status

287 {
288 // Write the data for the current year

290 results.variableUnit = VCardType::Unit();

292 .template buildAnnualSurveyReport<VCardType>(results, fileLevel, precision);

294 clusterIndex++;
295 }
296 }
