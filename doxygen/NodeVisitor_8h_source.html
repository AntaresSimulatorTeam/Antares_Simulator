35template<class RetT, class VisitorT, class NodeT, class... Args>
36RetT tryVisit(const Nodes::Node* node, VisitorT& visitor, Args... args)
37{
38 auto* x = dynamic_cast<const NodeT*>(node);
39 return visitor.visit(x, args...);
