174 void initializeFromArea(Data::Study* study, Data::Area* area)
175 {
176 // Get the number of years in parallel
177 pNbYearsParallel = study->maxNbYearsInParallel;
178 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
179
180 // Building the vector of group names the clusters belong to.
181 groupNames_ = sortedUniqueGroups(area->shortTermStorage.storagesByIndex);
182 groupToNumbers_ = giveNumbersToGroups(groupNames_);
183
184 nbColumns_ = groupNames_.size() * NB_COLS_PER_GROUP;
185
186 if (nbColumns_)
187 {
188 AncestorType::pResults.resize(nbColumns_);
189
190 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
191 {
192 pValuesForTheCurrentYear[numSpace] = new VCardType::IntermediateValuesDeepType
193 [nbColumns_];
194 }
195
196 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
197 {
198 for (unsigned int i = 0; i != nbColumns_; ++i)
199 {
200 pValuesForTheCurrentYear[numSpace][i].initializeFromStudy(*study);
201 }
202 }
203
204 for (unsigned int i = 0; i != nbColumns_; ++i)
205 {
206 AncestorType::pResults[i].initializeFromStudy(*study);
207 AncestorType::pResults[i].reset();
208 }
209 }
210 else
211 {
212 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
213 {
214 pValuesForTheCurrentYear[numSpace] = nullptr;
215 }
216
217 AncestorType::pResults.clear();
218 }
219 // Next
220 NextType::initializeFromArea(study, area);
221 }

238
239 void yearEnd(unsigned int year, unsigned int numSpace)
240 {
241 // Here we perform time-aggregations :
242 // ---------------------------------
243 // For a given MC year, from hourly results we compute daily, weekly, monthly and annual
244 // results by aggregation operations (averages or sums).
245 // Caution :
246 // - level results are stored in columns of which indices satisfy : col_index % 3 == 2.
247 // They are time-aggregated by means of averages
248 // - injection and withdrawal results are stored in columns of which indices
249 // satisfy : col_index % 3 != 2.
250 // They are time-aggregated by means of sums.
251
252 for (unsigned int column = 0; column < nbColumns_; column++)
253 {
254 switch (column % NB_COLS_PER_GROUP)
255 {
256 case VariableType::level:
257 pValuesForTheCurrentYear[numSpace][column]
258 .computeAveragesForCurrentYearFromHourlyResults();
259 break;
260 case VariableType::injection:
261 case VariableType::withdrawal:
262 pValuesForTheCurrentYear[numSpace][column].computeStatisticsForTheCurrentYear();
263 break;
264 }
265 }
266 // Next variable
267 NextType::yearEnd(year, numSpace);
268 }

270 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
271 unsigned int nbYearsForCurrentSummary)
272 {
273 // Here we compute synthesis :
274 // for each interval of any time period results (hourly, daily, weekly, ...),
275 // we compute the average over all MC years :
276 // For instance :
277 // - we compute the average of the results of the first hour over all MC years
278 // - or we compute the average of the results of the n-th day over all MC years
279 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
280 {
281 VariableAccessorType::ComputeSummary(pValuesForTheCurrentYear[numSpace],
282 AncestorType::pResults,
283 numSpaceToYear[numSpace]);
284 }
285
286 // Next variable
287 NextType::computeSummary(numSpaceToYear, nbYearsForCurrentSummary);
288 }

295 void hourForEachArea(State& state, unsigned int numSpace)
296 {
297 using namespace Antares::Data::ShortTermStorage;
298 const auto& shortTermStorage = state.area->shortTermStorage;
299
300 uint clusterIndex = 0;
301 for (const auto& cluster: shortTermStorage.storagesByIndex)
302 {
303 unsigned int groupNumber = groupToNumbers_[cluster.properties.groupName];
304 const auto& result = state.hourlyResults->ShortTermStorage[state.hourInTheWeek];
305 // Injection
306 pValuesForTheCurrentYear[numSpace][NB_COLS_PER_GROUP * groupNumber
307 + VariableType::injection][state.hourInTheYear]
308 += result.injection[clusterIndex];
309
310 // Withdrawal
311 pValuesForTheCurrentYear[numSpace][NB_COLS_PER_GROUP * groupNumber
312 + VariableType::withdrawal][state.hourInTheYear]
313 += result.withdrawal[clusterIndex];
314
315 // Levels
316 pValuesForTheCurrentYear[numSpace][NB_COLS_PER_GROUP * groupNumber
317 + VariableType::level][state.hourInTheYear]
318 += result.level[clusterIndex];
319
320 clusterIndex++;
321 }
322
323 // Next variable
324 NextType::hourForEachArea(state, numSpace);
325 }

373 void localBuildAnnualSurveyReport(SurveyResults& results,
374 int fileLevel,
375 int precision,
376 unsigned int numSpace) const
377 {
378 // Initializing external pointer on current variable non applicable status
379 results.isCurrentVarNA = AncestorType::isNonApplicable;
380
381 if (!AncestorType::isPrinted[0])
382 {
383 return;
384 }
385
386 for (unsigned int column = 0; column < nbColumns_; column++)
387 {
388 results.variableCaption = caption(column);
389 results.variableUnit = unit(column);
390 pValuesForTheCurrentYear[numSpace][column]
391 .template buildAnnualSurveyReport<VCardType>(results, fileLevel, precision);
392 }
393 }
394
395 void buildSurveyReport(SurveyResults& results,
396 int dataLevel,
397 int fileLevel,
398 int precision) const
399 {
400 // Building synthesis results
401 // ------------------------------
402
403 if (AncestorType::isPrinted[0])
404 {
405 // And only if we match the current data level _and_ precision level
406 if ((dataLevel & VCardType::categoryDataLevel)
407 && (fileLevel & VCardType::categoryFileLevel) && (precision & VCardType::precision))
408 {
409 results.isCurrentVarNA[0] = AncestorType::isNonApplicable[0];
410
411 for (unsigned int column = 0; column < nbColumns_; column++)
412 {
413 results.variableCaption = caption(column);
414 results.variableUnit = unit(column);
415 AncestorType::pResults[column]
416 .template buildSurveyReport<ResultsType, VCardType>(
417 results,
418 AncestorType::pResults[column],
419 dataLevel,
420 fileLevel,
421 precision);
422 }
423 }
424 }
425 // Ask to the next item in the static list to export its results as well
426 NextType::buildSurveyReport(results, dataLevel, fileLevel, precision);
427 }
