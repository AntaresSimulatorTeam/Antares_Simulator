34template<class T>
35Button::Button(wxWindow* parent,
36 const wxString& caption,
37 const char* bitmap,
38 T* object,
39 void (T::*method)(void*)) :
40 Panel(parent),
41 pCaption(caption),
42 pUserData(NULL),
43 pIcon(NULL),
44 pClickBehavior(clkDefault),
45 pRecommendedWidth(0),
46 pMiddleWidth(0),
47 pSelected(false),
48 pPushed(false),
49 pAutoToggle(false),
50 pBold(false),
51 pHover(true)
52{
53 // callback
54 using MethodType = void (T::*)(void*);
55 if (object)
56 pOnClick.bind(const_cast<T*>(object), reinterpret_cast<MethodType>(method));
57
58 SetBackgroundStyle(wxBG_STYLE_CUSTOM); // Required by both GTK and Windows
59 SetBackgroundColour(parent->GetBackgroundColour());
60 loadIconFromResource(bitmap);
61 precalculateCoordinates();
62}
