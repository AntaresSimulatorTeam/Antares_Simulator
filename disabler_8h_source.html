34template<class WindowT>
35class Disabler final
36{
37public:
38 Disabler(WindowT& p) : pWindow(p)
39 {
40 if (!pWindow.pUpdateCountLocker++)
41 pWindow.Enable(false);
42 }
43
44 ~Disabler()
45 {
46 assert(pWindow.pUpdateCountLocker > 0);
47 if (!(--pWindow.pUpdateCountLocker))
48 {
49 pWindow.Enable(true);
50 pWindow.SetFocus();
51 }
52 }
53
54private:
55 WindowT& pWindow;
56};
57
58template<>
59class Disabler<ApplWnd> final
60{
61public:
62 Disabler(ApplWnd& p) : pWindow(p)
63 {
64 if (!pWindow.pUpdateCountLocker++)
65 {
66 // disable the background timer
67 pWindow.backgroundTimerStop();
68 // disable the drag-drop target
69 pWindow.SetDropTarget(nullptr);
70 }
71 }
72
73 ~Disabler()
74 {
75 assert(pWindow.pUpdateCountLocker > 0);
76 if (!(--pWindow.pUpdateCountLocker))
77 {
78 // resetting the text of the status bar
79 pWindow.resetDefaultStatusBarText();
80 // The Form may have lost its focus
81 // This action must be delayed
82 pWindow.delayForceFocus();
83 // restore the drag-drop
84 pWindow.resetDragAndDrop();
85 }
86 }
87
88private:
90 ApplWnd& pWindow;
91};
