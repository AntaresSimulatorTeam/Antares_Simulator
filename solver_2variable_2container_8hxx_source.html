249template<class NextT>
250void List<NextT>::buildSurveyReport(SurveyResults& results,
251 int dataLevel,
252 int fileLevel,
253 int precision) const
254{
255 // Reset
256 results.data.columnIndex = 0;
257
258 // The new filename
259 results.data.filename.clear();
260 results.data.filename << results.data.output << SEP;
261 Category::FileLevelToStream(results.data.filename, fileLevel);
262 results.data.filename << '-';
263 Category::PrecisionLevelToStream(results.data.filename, precision);
264 results.data.filename << ".txt";
265
266 // Ask to all variables
267 NextType::buildSurveyReport(results, dataLevel, fileLevel, precision);
268
269 // If the column index is still equals to 0, that would mean we have nothing
270 // to do (there is no data to write)
271 if (results.data.columnIndex > 0)
272 results.saveToFile(dataLevel, fileLevel, precision);
273}

316template<class NextT>
317void List<NextT>::exportSurveyResults(bool global,
318 const Yuni::String& output,
319 unsigned int numSpace,
320 IResultWriter& writer)
321{
322 using namespace Antares;
323
324 // Infos
325 if (global)
326 {
327 logs.info(); // empty line
328 logs.checkpoint() << "Exporting the survey results...";
329 }
330 else
331 logs.info() << "Exporting the annual results";
332
333 auto survey = std::make_shared<SurveyResults>(*pStudy, output, writer);
334
335 // Year by year ?
336 survey->yearByYearResults = !global;
337
338 if (global)
339 {
340 // alias to the type of the report builder
341 using Builder = SurveyReportBuilder<true, ListType>;
342 // Building the survey results for each possible state
343 Builder::Run(*this, *survey);
344
345 // Exporting the Grid (information about the study)
346 survey->exportGridInfos();
347
348 // Exporting the digest
349 // The digest must be exported after the real report because some values
350 // are computed at this moment.
351 Builder::RunDigest(*this, *survey, writer);
352 }
353 else
354 {
355 // alias to the type of the report builder
356 using Builder = SurveyReportBuilder<false, ListType>;
357 // Building the survey results for each possible state
358 Builder::Run(*this, *survey, numSpace);
359 }
360}
