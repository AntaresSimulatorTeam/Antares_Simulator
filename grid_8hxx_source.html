56template<class NodeT>
57typename Grid<NodeT>::NodeP Grid<NodeT>::addNode(NodeT& n, std::string id)
58{
59 auto nodeIT
60 = std::find_if(pNodesList.begin(), pNodesList.end(), [&id](const NodeP& nodeP) -> bool {
61 return nodeP->getName() == id;
62 });
63
64 if (nodeIT == pNodesList.end())
65 {
66 NodeP newNode = new Graph::Node<NodeT>(&n);
67 newNode->setName(id);
68 pNodesList.push_back(newNode);
69 return newNode;
70 }
71 else
72 {
73 return *nodeIT;

77template<class NodeT>
78typename Grid<NodeT>::EdgeP Grid<NodeT>::addEdge(NodeP n1, NodeP n2, long weight)
79{
80 std::string s1(n1->getName());
81 std::string s2(n2->getName());
82 EdgeP edgePtr = findEdgeFromNodeNames(s1, s2);
83 if (edgePtr == nullptr)
84 {
85 EdgeP newEdge = new Edge<NodeT>(n1, n2);
86 inf += weight;
87 newEdge->setWeight(weight);
88
89 // add edge to the list
90 pEdgesList.push_back(newEdge);
91
92 // redefine inf
93 inf += weight;
94
95 // update adjency
96 adjency[newEdge->getOrigin()].insert(std::make_pair(newEdge->getDestination(), newEdge));
97 adjency[newEdge->getDestination()].insert(std::make_pair(newEdge->getOrigin(), newEdge));
98
99 return newEdge;
100 }
101 else
102 return edgePtr;
103}

105template<class NodeT>
106uint Grid<NodeT>::getNumberOfConnectedComponents()
107{
108 // create union-find set
109 MapNodes uf;
110
111 for (auto i = pNodesList.begin(); i != pNodesList.end(); i++)
112 {
113 uf.insert(std::make_pair(*i, *i));
114 }
115
116 // algorithm
117 for (auto i = pEdgesList.begin(); i != pEdgesList.end(); i++)
118 {
119 // if the two areas are not yet connected
120 if (uf[(*i)->getOrigin()] != uf[(*i)->getDestination()])
121 {
122 // update uf
123 NodeP cset(uf[(*i)->getDestination()]);
124 for (auto j = pNodesList.begin(); j != pNodesList.end(); j++)
125 {
126 if (uf[*j] == cset)
127 {
128 uf[*j] = uf[(*i)->getOrigin()];
129 }
130 }
131 }
132 }
133
134 std::unordered_map<NodeP, bool> cVec;
135 for (auto cIT = uf.begin(); cIT != uf.end(); cIT++)
136 {
137 cVec[cIT->second] = true;
138 }
139
140 return (uint)(cVec.size());
141}

143template<class NodeT>
144void Grid<NodeT>::kruskal()
145{
146 // clear spanning tree
147 pMinSpanningTree.clear();
148
149 // create union-find set
150 MapNodes uf;
151
152 for (auto i = pNodesList.begin(); i != pNodesList.end(); i++)
153 {
154 uf.insert(std::make_pair(*i, *i));
155 }
156
157 // create temporary sorted vector of Link
158 VectorEdgeP tempEdgesList = pEdgesList;
159 std::stable_sort(
160 tempEdgesList.begin(), tempEdgesList.end(), typename Graph::Edge<NodeT>::compareWeight());
161
162 // algorithm
163 for (auto i = tempEdgesList.begin(); i != tempEdgesList.end(); i++)
164 {
165 // if the two areas are not yet connected
166 if (uf[(*i)->getOrigin()] != uf[(*i)->getDestination()])
167 {
168 pMinSpanningTree.push_back(*i);
169
170 // update uf
171 NodeP cset(uf[(*i)->getDestination()]);
172 for (auto j = pNodesList.begin(); j != pNodesList.end(); j++)
173 {
174 if (uf[*j] == cset)
175 {
176 uf[*j] = uf[(*i)->getOrigin()];
177 }
178 }
179 }
180 }
181}

183template<class NodeT>
184bool Grid<NodeT>::buildMesh()
185{
186 // build the minimum spanning tree
187 kruskal();
188
189 if (pMinSpanningTree.empty())
190 return false;
191
192 // clear the spanning cycle base
193 pMesh.clear();
194 meshIndexMatrix.clear();
195
196 // initialization
197 std::vector<EdgeP> linksToBeAdded;
198 for (auto i = pEdgesList.begin(); i != pEdgesList.end(); i++)
199 {
200 // check if the link already belong to the skeleton
201 if (std::find(pMinSpanningTree.begin(), pMinSpanningTree.end(), (*i))
202 == pMinSpanningTree.end())
203 linksToBeAdded.push_back(*i);
204 }
205
206 std::vector<EdgeIncidence> incidenceMatrix(linksToBeAdded.size());
207 for (uint i = 0; i < linksToBeAdded.size(); i++)
208 {
209 incidenceMatrix[i] = getIncidenceVector(linksToBeAdded[i]);
210 }
211
212 for (uint i = 0; i < linksToBeAdded.size(); i++)
213 {
214 logs.info() << "Searching basis (loop " << i + 1 << "/" << linksToBeAdded.size() << ")";
215 // build the graph with two copies (+/-) for each node and edge
216 Grid<NodeT> polarisedDuplicate;
217 getDuplicatedGrid(polarisedDuplicate);
218 VectorEdgeP v = getEdgeVectorFromIncidence(incidenceMatrix[i]);
219 VectorEdgeP Ci;
221 VectorNodeP adjacentNodes;
222 // retrait des arêtes (u,v) qui sont dans Ei et remplacement dans le graphe dupliqué
223 // par(u+,v-) et (u-,v+)
224 for (uint j = 0; j < v.size(); j++)
225 {
226 if (std::find(adjacentNodes.begin(), adjacentNodes.end(), v[j]->getOrigin())
227 == adjacentNodes.end())
228 adjacentNodes.push_back(v[j]->getOrigin());
229 if (std::find(adjacentNodes.begin(), adjacentNodes.end(), v[j]->getDestination())
230 == adjacentNodes.end())
231 adjacentNodes.push_back(v[j]->getDestination());
232
233 EdgeP ei = polarisedDuplicate.findEdgeFromNodeNames(
234 v[j]->getOrigin()->getName() + "+", v[j]->getDestination()->getName() + "+");
235
236 NodeP ni1 = polarisedDuplicate.findNodeFromName(v[j]->getOrigin()->getName() + "+");
237 NodeP ni2
238 = polarisedDuplicate.findNodeFromName(v[j]->getDestination()->getName() + "-");
239
240 polarisedDuplicate.addEdge(ni1, ni2, ei->getWeight());
241 polarisedDuplicate.removeEdge(ei);
242
243 ei = polarisedDuplicate.findEdgeFromNodeNames(v[j]->getOrigin()->getName() + "-",
244 v[j]->getDestination()->getName() + "-");
245
246 ni1 = polarisedDuplicate.findNodeFromName(v[j]->getOrigin()->getName() + "-");
247 ni2 = polarisedDuplicate.findNodeFromName(v[j]->getDestination()->getName() + "+");
248
249 polarisedDuplicate.addEdge(ni1, ni2, ei->getWeight());
250 polarisedDuplicate.removeEdge(ei);
251 }
252
254 v = polarisedDuplicate.twoLevelPath(adjacentNodes);
255
257 Ci.clear();
258 std::vector<int> edgeIndices;
259 for (typename VectorEdgeP::iterator e = v.begin(); e != v.end(); e++)
260 {
261 auto name1 = (*e)->getOrigin()->getName();
262 name1 = name1.substr(0, name1.length() - 1);
263 auto name2 = (*e)->getDestination()->getName();
264 name2 = name2.substr(0, name2.length() - 1);
265 EdgeP ei = findEdgeFromNodeNames(name1, name2);
266 Ci.push_back(ei);
267
268 auto eIT = std::find(pEdgesList.begin(), pEdgesList.end(), ei);
269 int pos = (int)std::distance(pEdgesList.begin(), eIT);
270 edgeIndices.push_back(pos);
271 }
272
273 pMesh.push_back(Ci);
274 EdgeIncidence I = getIncidenceVector(Ci);
275 meshIndexMatrix.push_back(edgeIndices);
276 // remplacement par la différence symétrique des des jeux d'arêtes
277 for (uint j = i + 1; j < linksToBeAdded.size(); j++)
278 {
279 if (incidenceInnerProduct(I, incidenceMatrix[j]) == 1)
280 {
281 incidenceMatrix[j] = incidenceXOR(incidenceMatrix[i], incidenceMatrix[j]);
282 }
283 }
284 }
285
286 return true;
287}

293template<class NodeT>
294bool Grid<NodeT>::getDuplicatedGrid(Grid<NodeT>& grid)
295{
296 // duplicate nodes
297 for (typename VectorNodeP::iterator node = pNodesList.begin(); node != pNodesList.end(); node++)
298 {
299 { //+
300 grid.addNode((*(*node)->nodeProperties), (*node)->getName() + "+");
301 }
302
303 { //-
304 grid.addNode((*(*node)->nodeProperties), (*node)->getName() + "-");
305 }
306 }
307
308 // Duplicate edges
309 for (typename VectorEdgeP::iterator e = pEdgesList.begin(); e != pEdgesList.end(); e++)
310 {
311 { //+
312 auto nodeOrigIT = std::find_if(
313 grid.pNodesList.begin(), grid.pNodesList.end(), [&e](const NodeP& nodeP) -> bool {
314 return nodeP->getName() == (*e)->getOrigin()->getName() + "+";
315 });
316
317 auto nodeDestIT = std::find_if(
318 grid.pNodesList.begin(), grid.pNodesList.end(), [&e](const NodeP& nodeP) -> bool {
319 return nodeP->getName() == (*e)->getDestination()->getName() + "+";
320 });
321
322 grid.addEdge(*nodeOrigIT, *nodeDestIT, (*e)->getWeight());
323 }
324
325 { //-
326 auto nodeOrigIT = std::find_if(
327 grid.pNodesList.begin(), grid.pNodesList.end(), [&e](const NodeP& nodeP) -> bool {
328 return nodeP->getName() == (*e)->getOrigin()->getName() + "-";
329 });
330
331 auto nodeDestIT = std::find_if(
332 grid.pNodesList.begin(), grid.pNodesList.end(), [&e](const NodeP& nodeP) -> bool {
333 return nodeP->getName() == (*e)->getDestination()->getName() + "-";
334 });
335
336 grid.addEdge(*nodeOrigIT, *nodeDestIT, (*e)->getWeight());
337 }
338 }
339 return true;
340}

348{

350 for (typename VectorNodeP::iterator node = pNodesList.begin(); node != pNodesList.end(); node++)

377template<class NodeT>
378typename Grid<NodeT>::VectorEdgeP Grid<NodeT>::twoLevelPath(VectorNodeP vN)
379{
380 VectorEdgeP SP, minSP;
381 double minLength = DBL_MAX, length;
382 for (typename VectorNodeP::iterator n = vN.begin(); n != vN.end(); n++)
383 {
384 SP = findShortestPath(findNodeFromName((*n)->getName() + "+"),
385 findNodeFromName((*n)->getName() + "-"));
386 length = std::accumulate(
387 SP.begin(), SP.end(), (long)0, typename Graph::Edge<NodeT>::addpWeight());
388 if (length < minLength)
389 {
390 minLength = length;
391 minSP = SP;
392 }
393 }
394
395 return minSP;
396}

399template<class NodeT>
400typename Grid<NodeT>::VectorEdgeP Grid<NodeT>::findShortestPath(NodeP node1, NodeP node2) const
401{
402 assert(node1 != node2);
403
404 // Dijkstra (lots of computation which could be optimised)
405 // initialization
406 std::map<NodeP, double> dist;
407 std::map<NodeP, NodeP> prev;
408
409 //decorate to avoid warnings at compile
410 [[maybe_unused]] bool checkNode1(false), checkNode2(false);
411 VectorNodeP nodes = pNodesList;
412
413 for (auto i = nodes.begin(); i != nodes.end(); i++)
414 {
415 dist[(*i)] = inf;
416 prev[(*i)] = 0;
417 if ((*i) == node1)
418 {
419 checkNode1 = true;
420 }
421 else if ((*i) == node2)
422 {
423 checkNode2 = true;
424 }
425 }
426 assert(checkNode1 && checkNode2);
427 dist[node1] = 0;
428 // search for shortest path
429 bool targetFound(0);
430 while (!targetFound)
431 {
432 // look for closest element u belonging in nodes
433 std::pair<Grid::NodeP, double> u(nullptr, inf - 1);
434 typename Grid::VectorNodeP::iterator ui;
435 for (auto i = nodes.begin(); i != nodes.end(); i++)
436 {
437 if (dist[(*i)] < u.second)
438 {
439 u.first = (*i);
440 u.second = dist[(*i)];
441 ui = i;
442 }
443 }
444
445 // see if destination node has been f
446 if (u.first == node2)
447 {
448 targetFound = true;
449 break;
450 }
451
452 // remove u from the list of nodes
453 nodes.erase(ui);
454
455 // update the neighoubours of u
456 for (auto i = adjency.at(u.first).begin(); i != adjency.at(u.first).end(); i++)
457 {
458 if (i->second == nullptr)
459 assert(0);
460
461 if ((u.second + i->second->getWeight()) < dist[i->first])
462 {
463 dist[i->first] = u.second + i->second->getWeight();
464 prev[i->first] = u.first;
465 }
466 }
467 }
468
469 // rebuild path
470 Grid::VectorEdgeP path;
471 Grid::NodeP currentNode = node2;
472
473 while (!prev[currentNode] == 0)
474 {
475 path.push_back(adjency.at(currentNode).at(prev[currentNode]));
476 currentNode = prev[currentNode];
477 }
478
479 return path;
480}
