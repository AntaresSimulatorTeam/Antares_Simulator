65template<class U>
66IniFile::Property* IniFile::Section::add(const AnyString& key, const U& value)
67{
68 auto* p = new Property(key, value);
69 if (!lastProperty)
70 firstProperty = p;
71 else
72 lastProperty->next = p;
73 lastProperty = p;
74 return p;
75}
