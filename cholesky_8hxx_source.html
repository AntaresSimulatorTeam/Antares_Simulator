29#define ANTARES_CHOLESKY_EPSIMIN ((T)1.0e-9)
30
31namespace Antares
32{
33namespace Solver
34{
35template<class T, class U1, class U2>
36bool Cholesky(U1& L, U2& A, uint size, T* temp)
37{
38 using namespace Yuni;
39
40 for (uint i = 0; i != size; ++i)
41 temp[i] = 0;
42
43 T som;
44
45 for (uint i = 0; i < size; ++i)
46 {
47 typename MatrixSubColumn<U1>::Type Li = L[i];
48
49 // on calcule d'abord L[i][i]
50 som = A[i][i];
51 for (int j = 0; j <= (int)(i - 1); ++j)
52 som -= Li[j] * Li[j];
53
54 if (som > ANTARES_CHOLESKY_EPSIMIN)
55 {
56 Li[i] = Math::SquareRootNoCheck(som);
57
58 // maintenant on cherche L[k][i], k > i.
59 for (uint k = i + 1; k < size; ++k)
60 {
61 typename MatrixSubColumn<U1>::Type Lk = L[k];
62 typename MatrixSubColumn<U2>::Type Ak = A[k];
63
64 if (temp[k] == Ak[k])
65 {
66 Lk[i] = 0;
67 }
68 else
69 {
70 som = Ak[i];
71 for (int j = 0; j <= (int)(i - 1); ++j)
72 som -= Li[j] * Lk[j];
73
74 Lk[i] = som / Li[i];
75 temp[k] += Lk[i] * Lk[i];
76
77 // si temp[k] = A[k][k] la matrice n'est pas dp mais il est encore possible
78 // qu'elle soit sdp.
79 // si temp > A[k][k] alors il est certain que A n'est ni sdp ni dp donc on
80 // arrete le calcul
81 if (temp[k] > Ak[k])
82 return true;
83 }
84 }
85 }
86 else
87 {
88 // annule le reste de la colonne
89 for (uint k = i; k != size; ++k)
90 L[k][i] = 0;
91 }
92 }
93
94 return false;
95}

97} // namespace Solver
98} // namespace Antares

100#undef ANTARES_CHOLESKY_EPSIMIN
