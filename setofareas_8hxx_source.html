50template<class NextT>
51void SetsOfAreas<NextT>::initializeFromStudy(Data::Study& study)
52{
53 using namespace Antares;
54
55 // The study
56 pStudy = &study;
57 // alias to the set of sets of areas
58 auto& sets = study.setsOfAreas;
59 // Reserving the memory
60 pSetsOfAreas.reserve(sets.size());
61 pOriginalSets.reserve(sets.size());
62
63 // For each set...
64 for (uint setIndex = 0; setIndex != sets.size(); ++setIndex)
65 {
66 if (!sets.hasOutput(setIndex))
67 continue;
68 // Name of the set
69 const auto& setname = sets.caption(setIndex);
70
71 // Useless if the result set is empty
72 if (!sets.resultSize(setIndex))
73 {
74 logs.warning() << "The set of areas named '" << setname
75 << "' is empty. No output will be produced for this set.";
76 continue;
77 }
78
79 // Instancing a new set of variables of the area
80 NextType* n = new NextType();
81
82 // Initialize the variables
83 // From the study
84 n->initializeFromStudy(study);
85
86 // Making specific variables non applicable in following output reports :
87 // - annual district reports
88 // - over all years district statistics reports
89 n->broadcastNonApplicability(true);
90
91 // For each current set's variable, getting the print status, that is :
92 // is variable's column(s) printed in output (set of areas) reports ?
93 n->getPrintStatusFromStudy(study);
94
95 // Adding the variables for the area in the list
96 pSetsOfAreas.push_back(n);
97 auto* originalSet = &sets[setIndex];
98 assert(originalSet != NULL);
99 assert(!originalSet->empty());
100 pOriginalSets.push_back(originalSet);
101
102 pNames.push_back(setname);
103 }
104
105 // Initializing iterators
106 pBegin = pSetsOfAreas.begin();
107 pEnd = pSetsOfAreas.end();
108}

242template<class NextT>
243void SetsOfAreas<NextT>::buildDigest(SurveyResults& results, int digestLevel, int dataLevel) const
244{
245 int count_int = count;
246 bool setOfAreasDataLevel = dataLevel & Category::setOfAreas;
247 if (count_int && setOfAreasDataLevel)
248 {
249 // Reset
250 results.data.rowCaptions.clear();
251 results.data.rowCaptions.resize(pSetsOfAreas.size());
252 results.data.area = nullptr;
253 results.data.rowIndex = 0;
254
255 for (auto i = pBegin; i != pEnd; ++i)
256 {
257 results.data.columnIndex = 0;
258 results.data.rowCaptions[results.data.rowIndex].clear()
259 << "@ " << pNames[results.data.rowIndex];
260 (*i)->buildDigest(results, digestLevel, dataLevel);
261 ++results.data.rowIndex;
262 }
263 }
264}

278{
