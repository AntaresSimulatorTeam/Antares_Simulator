229 void hourForEachLink(State& state, unsigned int numSpace)
230 {
231 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] = Yuni::Math::Abs(
232 state.problemeHebdo->VariablesDualesDesContraintesDeNTC[state.hourInTheWeek]
233 .VariableDualeParInterconnexion[state.link->index]);
234
235 // This value should be reset to zero if (flow_lowerbound) < flow < (flow_upperbound) (with
236 // signed values)
237 double flow
238 = state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek].ValeurDuFlux[state.link->index];
239 double flow_lowerbound = -state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
240 .ValeurDeNTCExtremiteVersOrigine[state.link->index];
241 double flow_upperbound = state.problemeHebdo->ValeursDeNTC[state.hourInTheWeek]
242 .ValeurDeNTCOrigineVersExtremite[state.link->index];
243
244 if (flow - 0.001 > flow_lowerbound && flow + 0.001 < flow_upperbound)
245 pValuesForTheCurrentYear[numSpace][state.hourInTheYear] = 0.;
246
247 // Next item in the list
248 NextType::hourForEachLink(state, numSpace);
249 }
