76template<enum Data::TimeSeriesType T>
77inline bool XCastCoefficients<T>::cellValue(int x, int y, const Yuni::String& value)
78{
79 using namespace Antares::Data;
80
81 if (!pArea)
82 return false;
83 auto& xcast = *(pArea->xcastData<T>());
84 float v;
85 if (!value.to(v))
86 return false;
87 if ((uint)x >= xcast.data.width || (uint)y >= xcast.data.height)
88 return false;
89
90 switch (x)
91 {
92 case XCast::dataCoeffAlpha:
93 {
94 switch (xcast.distribution)
95 {
96 case XCast::dtUniform:
97 break;
98 case XCast::dtBeta:
99 {
100 if (v <= 0.f)
101 return MatrixAncestorType::cellValue(x, y, "0");
102 break;
103 }
104 case XCast::dtNormal:
105 break;
106 default:
107 {
108 if (v < 1.f || v > 50.f)
109 return MatrixAncestorType::cellValue(x, y, "1");
110 }
111 }
112 break;
113 }
114 case XCast::dataCoeffBeta:
115 {
116 switch (xcast.distribution)
117 {
118 case XCast::dtUniform:
119 break;
120 default:
121 {
122 if (v <= 0.f)
123 return MatrixAncestorType::cellValue(x, y, "0");
124 break;
125 }
126 }
127 break;
128 }
129 case XCast::dataCoeffGamma:
130 {
131 switch (xcast.distribution)
132 {
133 case XCast::dtUniform:
134 {
135 // if (v > xcast.data.entry[XCast::dataCoeffDelta][y])
136 // return MatrixAncestorType::cellValue(x, y,
137 // String(xcast.data.entry[XCast::dataCoeffDelta][y]));
138 break;
139 }
140 case XCast::dtBeta:
141 {
142 // if (v >= xcast.data.entry[XCast::dataCoeffDelta][y])
143 // return MatrixAncestorType::cellValue(x, y, "0");
144 break;
145 }
146 case XCast::dtNormal:
147 break;
148 default:
149 break;
150 }
151 break;
152 }
153 case XCast::dataCoeffDelta:
154 {
155 switch (xcast.distribution)
156 {
157 case XCast::dtUniform:
158 {
159 // if (v < xcast.data.entry[XCast::dataCoeffGamma][y])
160 // return MatrixAncestorType::cellValue(x, y,
161 // xcast.data.entry[XCast::dataCoeffGamma][y]);
162 break;
163 }
164 case XCast::dtBeta:
165 {
166 // if (v <= xcast.data.entry[XCast::dataCoeffGamma][y])
167 // return MatrixAncestorType::cellValue(x, y, "1");
168 break;
169 }
170 default:
171 break;
172 }
173 break;
174 }
175
176 case XCast::dataCoeffTheta:
177 {
178 if (v < 0.f)
179 return MatrixAncestorType::cellValue(x, y, "0");
180 break;
181 }
182 case XCast::dataCoeffMu:
183 {
184 if (v < 1.f)
185 return MatrixAncestorType::cellValue(x, y, "1");
186 if (v > 23.f)
187 return MatrixAncestorType::cellValue(x, y, "23");
188 break;
189 }
190 }
191 return MatrixAncestorType::cellValue(x, y, value);
192}

207template<enum Data::TimeSeriesType T>
208IRenderer::CellStyle XCastCoefficients<T>::cellStyle(int col, int row) const
209{
210 // law \ coeff alpha beta gamma delta
211 //
212 // uniform ? ? R R
213 // beta >0 >0 R R
214 // normal R >0 R R
215 // weibull 1 <= x <= 50 >0 R ?
216 // gamma 1 <= x <= 50 >0 R ?
217 //
218 // R : -inf < x < +inf
219 // ? : whatever (disabled)
220 //
221 // mu : 1 <= mu <= 23
222 // if uniform || beta then gamma < delta
223 //
224 using namespace Antares::Data;
225 if (!pArea)
226 return IRenderer::cellStyleDefault;
227 XCast& xcast = *(pArea->xcastData<T>());
228 if ((uint)col >= xcast.data.width || (uint)row >= xcast.data.height)
229 return IRenderer::cellStyleDefault;
230
231 float d = xcast.data[col][row];
232 switch (col)
233 {
234 case XCast::dataCoeffAlpha:
235 {
236 switch (xcast.distribution)
237 {
238 case XCast::dtUniform:
239 return IRenderer::cellStyleDefaultDisabled;
240 case XCast::dtBeta:
241 {
242 if (d <= 0.f)
243 return IRenderer::cellStyleError;
244 break;
245 }
246 case XCast::dtNormal:
247 break;
248 default:
249 {
250 if (d < 1.f || d > 50.f)
251 return IRenderer::cellStyleError;
252 }
253 }
254 break;
255 }
256 case XCast::dataCoeffBeta:
257 {
258 switch (xcast.distribution)
259 {
260 case XCast::dtUniform:
261 return IRenderer::cellStyleDefaultDisabled;
262 case XCast::dtNormal:
263 {
264 if (d < 0.)
265 return IRenderer::cellStyleError;
266 break;
267 }
268 default:
269 {
270 if (d <= 0.)
271 return IRenderer::cellStyleError;
272 break;
273 }
274 }
275 break;
276 }
277 case XCast::dataCoeffGamma:
278 {
279 switch (xcast.distribution)
280 {
281 case XCast::dtUniform:
282 {
283 if (d > xcast.data.entry[XCast::dataCoeffDelta][row])
284 return IRenderer::cellStyleError;
285 break;
286 }
287 case XCast::dtBeta:
288 {
289 if (d >= xcast.data.entry[XCast::dataCoeffDelta][row])
290 return IRenderer::cellStyleError;
291 break;
292 }
293 case XCast::dtNormal:
294 return IRenderer::cellStyleDefaultDisabled;
295 default:
296 break;
297 }
298 break;
299 }
300 case XCast::dataCoeffDelta:
301 {
302 switch (xcast.distribution)
303 {
304 case XCast::dtUniform:
305 {
306 if (d < xcast.data.entry[XCast::dataCoeffGamma][row])
307 return IRenderer::cellStyleError;
308 break;
309 }
310 case XCast::dtBeta:
311 {
312 if (d <= xcast.data.entry[XCast::dataCoeffGamma][row])
313 return IRenderer::cellStyleError;
314 break;
315 }
316 default:
317 return IRenderer::cellStyleDefaultDisabled;
318 }
319 break;
320 }
321 case XCast::dataCoeffTheta:
322 {
323 if (d < 0.f)
324 return IRenderer::cellStyleError;
325 break;
326 }
327 case XCast::dataCoeffMu:
328 {
329 if (d < 1.f || d > 23.f)
330 return IRenderer::cellStyleError;
331 break;
332 }
333 default:
334 break;
335 }
336 return Yuni::Math::Zero(d) ? IRenderer::cellStyleDefaultDisabled : IRenderer::cellStyleDefault;
337}
