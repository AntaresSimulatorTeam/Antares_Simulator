102template<class NextT = Container::EndOfList>
103class PSP : public Variable::IVariable<PSP<NextT>, NextT, VCardPSP>
104{
105public:
107 typedef NextT NextType;
109 typedef VCardPSP VCardType;
111 typedef Variable::IVariable<PSP<NextT>, NextT, VCardType> AncestorType;
112
114 typedef typename VCardType::ResultsType ResultsType;
115
116 typedef VariableAccessor<ResultsType, VCardType::columnCount> VariableAccessorType;
117
118 enum
119 {
121 count = 1 + NextT::count,
122 };
123
124 template<int CDataLevel, int CFile>
125 struct Statistics
126 {
127 enum
128 {
129 count
130 = ((VCardType::categoryDataLevel & CDataLevel && VCardType::categoryFileLevel & CFile)
131 ? (NextType::template Statistics<CDataLevel, CFile>::count
132 + VCardType::columnCount * ResultsType::count)
133 : NextType::template Statistics<CDataLevel, CFile>::count),
134 };
135 };
136
137public:
138 ~PSP()
139 {
140 delete[] pValuesForTheCurrentYear;
141 }
142
143 void initializeFromStudy(Data::Study& study)
144 {
145 pNbYearsParallel = study.maxNbYearsInParallel;
146
147 // Average on all years
148 InitializeResultsFromStudy(AncestorType::pResults, study);
149
150 // Intermediate values
151 pValuesForTheCurrentYear = new VCardType::IntermediateValuesBaseType[pNbYearsParallel];
152 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
153 pValuesForTheCurrentYear[numSpace].initializeFromStudy(study);
154
155 // Next
156 NextType::initializeFromStudy(study);
157 }
158
159 void initializeFromArea(Data::Study* study, Data::Area* area)
160 {
161 // Copy raw values
162 for (unsigned int numSpace = 0; numSpace < pNbYearsParallel; numSpace++)
163 (void)::memcpy(pValuesForTheCurrentYear[numSpace].hour,
164 area->miscGen.entry[Data::fhhPSP],
165 sizeof(double) * area->miscGen.height);
166
167 // Next
168 NextType::initializeFromArea(study, area);
169 }
170
171 void initializeFromLink(Data::Study* study, Data::AreaLink* link)
172 {
173 // Next
174 NextType::initializeFromAreaLink(study, link);
175 }

211 // Next variable

214
215 void computeSummary(std::map<unsigned int, unsigned int>& numSpaceToYear,
216 unsigned int nbYearsForCurrentSummary)
217 {
218 for (unsigned int numSpace = 0; numSpace < nbYearsForCurrentSummary; ++numSpace)
219 AncestorType::pResults.merge(numSpaceToYear[numSpace],
220 pValuesForTheCurrentYear[numSpace]);
221
222 // Next variable
223 NextType::computeSummary(numSpaceToYear, nbYearsForCurrentSummary);
224 }

238 Antares::Memory::Stored<double>::ConstReturnType retrieveRawHourlyValuesForCurrentYear(
239 unsigned int,

245 void localBuildAnnualSurveyReport(SurveyResults& results,
246 int fileLevel,
247 int precision,
248 unsigned int numSpace) const
249 {
250 // Initializing external pointer on current variable non applicable status
251 results.isCurrentVarNA = AncestorType::isNonApplicable;
252
253 if (AncestorType::isPrinted[0])
254 {
255 // Write the data for the current year
256 results.variableCaption = VCardType::Caption();
257 results.variableUnit = VCardType::Unit();
258 pValuesForTheCurrentYear[numSpace].template buildAnnualSurveyReport<VCardType>(
259 results, fileLevel, precision);
260 }
261 }
